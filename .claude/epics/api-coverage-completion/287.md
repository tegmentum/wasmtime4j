# Task 287: Advanced Memory Management Features

## Overview

Implement advanced WebAssembly memory management features including multi-memory support, memory sharing, custom allocators, and memory growth callbacks to achieve complete memory API coverage.

## Problem Statement

Current memory implementation provides basic linear memory support but lacks advanced features:
- **Multi-Memory**: Cannot use multiple memory instances per module
- **Memory Sharing**: No support for shared memory between instances
- **Custom Allocators**: Cannot use custom memory allocation strategies
- **Growth Callbacks**: No notifications when memory grows
- **Memory Pools**: No memory pooling for performance optimization

This limits advanced use cases and prevents optimal memory utilization in complex WebAssembly applications.

### Phase 1: Multi-Memory Infrastructure (Week 1, Days 1-4)

#### Native Multi-Memory Support
```c
// JNI implementation for multi-memory support
JNIEXPORT jobjectArray JNICALL Java_Instance_getMemoriesNative(
    JNIEnv *env, jobject obj, jlong instance_ptr
) {
    wasmtime_instance_t *instance = (wasmtime_instance_t*)instance_ptr;

    // Get all memory exports
    wasm_exporttype_vec_t export_types;
    wasmtime_instance_export_types(instance, &export_types);

    // Count memory exports
    size_t memory_count = 0;
    for (size_t i = 0; i < export_types.size; i++) {
        if (wasm_exporttype_type(export_types.data[i])->kind == WASM_EXTERN_MEMORY) {
            memory_count++;
        }
    }

    // Create Java array for memory instances
    jclass memoryClass = (*env)->FindClass(env, "ai/tegmentum/wasmtime4j/WasmMemory");
    jobjectArray memories = (*env)->NewObjectArray(env, memory_count, memoryClass, NULL);

    // Populate array with memory instances
    size_t memory_index = 0;
    for (size_t i = 0; i < export_types.size; i++) {
        if (wasm_exporttype_type(export_types.data[i])->kind == WASM_EXTERN_MEMORY) {
            wasmtime_extern_t memory_extern;
            wasmtime_instance_export_get(instance, i, &memory_extern);

            jobject jmemory = createJavaMemory(env, &memory_extern.of.memory);
            (*env)->SetObjectArrayElement(env, memories, memory_index++, jmemory);
        }
    }

    wasm_exporttype_vec_delete(&export_types);
    return memories;
}
```

#### Panama Multi-Memory Implementation
```java
public class PanamaInstanceImpl implements Instance {
    private final MemorySegment instanceHandle;
    private final Map<String, WasmMemory> memories = new ConcurrentHashMap<>();

    @Override
    public Map<String, WasmMemory> getMemories() {
        if (memories.isEmpty()) {
            loadMemories();
        }
        return Collections.unmodifiableMap(memories);
    }

    private void loadMemories() {
        try (Arena arena = Arena.ofConfined()) {
            MemorySegment exportsPtr = arena.allocate(ADDRESS);
            MemorySegment exportsCount = arena.allocate(JAVA_INT);

            int result = WasmtimeNative.wasmtime_instance_exports(
                instanceHandle, exportsPtr, exportsCount);

            if (result != 0) {
                throw new WasmException("Failed to get instance exports");
            }

            MemorySegment exports = exportsPtr.get(ADDRESS, 0);
            int count = exportsCount.get(JAVA_INT, 0);

            for (int i = 0; i < count; i++) {
                MemorySegment export = exports.asSlice(i * EXPORT_SIZE, EXPORT_SIZE);

                if (isMemoryExport(export)) {
                    String name = getExportName(export);
                    WasmMemory memory = createMemoryFromExport(export);
                    memories.put(name, memory);
                }
            }
        }
    }
}
```

### Phase 2: Memory Sharing Implementation (Week 1, Days 5-7)

#### Shared Memory Implementation
```java
public class SharedMemoryImpl implements SharedMemory {
    private final WasmMemory underlyingMemory;
    private final MemorySharingMode sharingMode;
    private final Map<Long, MemoryView> threadViews = new ConcurrentHashMap<>();
    private final ReadWriteLock sharingLock = new ReentrantReadWriteLock();
    private final AtomicReference<MemorySharingStatistics> statistics =
        new AtomicReference<>(new MemorySharingStatistics(0, 0, 0, 0, 0));

    @Override
    public MemoryView createThreadView(long threadId) {
        Lock lock = sharingMode == MemorySharingMode.SHARED_WRITE
            ? sharingLock.writeLock()
            : sharingLock.readLock();

        lock.lock();
        try {
            if (threadViews.containsKey(threadId)) {
                throw new IllegalStateException("Thread view already exists: " + threadId);
            }

            MemoryView view = new MemoryViewImpl(this, threadId);
            threadViews.put(threadId, view);

            updateStatistics();
            return view;

        } finally {
            lock.unlock();
        }
    }

    @Override
    public void synchronize() throws WasmException {
        sharingLock.writeLock().lock();
        try {
            // Flush all pending writes and ensure memory consistency
            nativeSynchronizeSharedMemory(getHandle());

            // Update statistics
            MemorySharingStatistics current = statistics.get();
            statistics.set(new MemorySharingStatistics(
                current.activeReaders(),
                current.activeWriters(),
                current.totalReads(),
                current.totalWrites(),
                current.synchronizationCount() + 1
            ));

        } finally {
            sharingLock.writeLock().unlock();
        }
    }
}
```

### Phase 3: Custom Allocators and Pooling (Week 2)

#### Pooling Memory Allocator
```java
public class PoolingMemoryAllocator implements MemoryAllocator {
    private final Map<MemoryType, Queue<WasmMemory>> pools = new ConcurrentHashMap<>();
    private final int maxPoolSize;
    private final long maxMemorySize;
    private final AtomicReference<AllocationStatistics> statistics;

    @Override
    public WasmMemory allocate(MemoryType type) throws WasmException {
        Queue<WasmMemory> pool = pools.computeIfAbsent(type, k -> new ConcurrentLinkedQueue<>());

        WasmMemory memory = pool.poll();
        if (memory != null && memory.isValid()) {
            // Reset memory state for reuse
            resetMemory(memory);
            updateStatistics(false, true);
            return memory;
        }

        // No pooled memory available, create new
        memory = createNewMemory(type);
        updateStatistics(true, false);
        return memory;
    }

    @Override
    public void deallocate(WasmMemory memory) {
        MemoryType type = memory.getType();
        Queue<WasmMemory> pool = pools.get(type);

        if (pool != null && pool.size() < maxPoolSize &&
            memory.size() <= maxMemorySize) {
            // Return to pool for reuse
            pool.offer(memory);
        } else {
            // Pool full or memory too large, dispose immediately
            memory.close();
        }

        updateStatistics(false, false);
    }

    private void resetMemory(WasmMemory memory) {
        // Clear memory contents for security
        try (MemorySegment segment = memory.buffer()) {
            segment.fill((byte) 0);
        }
    }
}
```

```java
void testSharedMemory() {
    MemoryType sharedType = MemoryType.shared(1, Optional.of(10L));
    MemoryAllocator allocator = MemoryAllocators.simple();

    SharedMemory sharedMem = (SharedMemory) allocator.allocate(sharedType);
    assertEquals(MemorySharingMode.SHARED_WRITE, sharedMem.getSharingMode());

    // Create views for different threads
    MemoryView view1 = sharedMem.createThreadView(Thread.currentThread().getId());
    MemoryView view2 = sharedMem.createThreadView(Thread.currentThread().getId() + 1);

    // Test synchronized access
    CompletableFuture<Void> writer = CompletableFuture.runAsync(() -> {
        MemorySegment segment = view1.writableSlice(0, 4);
        segment.set(JAVA_INT, 0, 12345);
    });

    CompletableFuture<Integer> reader = CompletableFuture.supplyAsync(() -> {
        writer.join(); // Wait for write to complete
        sharedMem.synchronize();
        MemorySegment segment = view2.readOnlySlice(0, 4);
        return segment.get(JAVA_INT, 0);
    });

    assertEquals(12345, reader.join().intValue());
}
```
