#[no_mangle]
pub extern "C" fn wasmtime4j_wasi_read_file_async(
    wasi_ctx_ptr: *mut c_void,
    path: *const c_char,
    callback_id: usize
) -> bool {
    let wasi_ctx = unsafe { &mut *(wasi_ctx_ptr as *mut WasiCtx) };
    let path_str = unsafe { CStr::from_ptr(path).to_str().unwrap() };

    // Spawn async file read operation
    tokio::spawn(async move {
        match tokio::fs::read(path_str).await {
            Ok(data) => {
                notify_read_success(callback_id, data);
            }
            Err(error) => {
                notify_read_error(callback_id, error);
            }
        }
    });

    true
}

#[no_mangle]
pub extern "C" fn wasmtime4j_wasi_create_socket(
    wasi_ctx_ptr: *mut c_void,
    socket_type: i32,
    callback_id: usize
) -> bool {
    let wasi_ctx = unsafe { &mut *(wasi_ctx_ptr as *mut WasiCtx) };

    tokio::spawn(async move {
        let socket = match socket_type {
            0 => tokio::net::TcpSocket::new_v4(),
            1 => tokio::net::TcpSocket::new_v6(),
            _ => {
                notify_socket_error(callback_id, "Invalid socket type".to_string());
                return;
            }
        };

        match socket {
            Ok(sock) => {
                notify_socket_success(callback_id, sock);
            }
            Err(error) => {
                notify_socket_error(callback_id, error.to_string());
            }
        }
    });

    true
}
