// Enhanced JNI Engine with async operations
public class JniEngine implements Engine {

    // Existing methods...

    // Async module compilation
    @Override
    public CompletableFuture<Module> compileAsync(byte[] wasmBytes) {
        return compileAsync(wasmBytes, null);
    }

    @Override
    public CompletableFuture<Module> compileAsync(byte[] wasmBytes, CompilationProgress progress) {
        CompletableFuture<Module> future = new CompletableFuture<>();
        long callbackId = registerCompilationCallback(future, progress);

        if (nativeCompileAsync(enginePtr, wasmBytes, wasmBytes.length, callbackId)) {
            return future;
        } else {
            future.completeExceptionally(new WasmException("Failed to start async compilation"));
            return future;
        }
    }

    @Override
    public CompletableFuture<Module> compileInBackground(byte[] wasmBytes, CompilationPriority priority) {
        CompletableFuture<Module> future = new CompletableFuture<>();
        long callbackId = registerCompilationCallback(future, null);

        if (nativeCompileInBackground(enginePtr, wasmBytes, wasmBytes.length, priority.ordinal(), callbackId)) {
            return future;
        } else {
            future.completeExceptionally(new WasmException("Failed to start background compilation"));
            return future;
        }
    }

    // Callback management for async operations
    private static final AtomicLong CALLBACK_COUNTER = new AtomicLong(1);
    private static final ConcurrentHashMap<Long, CompilationCallback> COMPILATION_CALLBACKS = new ConcurrentHashMap<>();
    private static final ConcurrentHashMap<Long, FunctionCallback> FUNCTION_CALLBACKS = new ConcurrentHashMap<>();

    private long registerCompilationCallback(CompletableFuture<Module> future, CompilationProgress progress) {
        long callbackId = CALLBACK_COUNTER.getAndIncrement();
        COMPILATION_CALLBACKS.put(callbackId, new CompilationCallback(future, progress));
        return callbackId;
    }

    private long registerFunctionCallback(CompletableFuture<WasmValue> future, Store store) {
        long callbackId = CALLBACK_COUNTER.getAndIncrement();
        FUNCTION_CALLBACKS.put(callbackId, new FunctionCallback(future, store));
        return callbackId;
    }

    // Native callback implementations (called from native code)
    public static void onCompilationSuccess(long callbackId, long modulePtr) {
        CompilationCallback callback = COMPILATION_CALLBACKS.remove(callbackId);
        if (callback != null) {
            try {
                Module module = new JniModule(modulePtr);
                callback.future.complete(module);
            } catch (Exception e) {
                callback.future.completeExceptionally(e);
            }
        }
    }

    public static void onCompilationError(long callbackId, String errorMessage) {
        CompilationCallback callback = COMPILATION_CALLBACKS.remove(callbackId);
        if (callback != null) {
            callback.future.completeExceptionally(new WasmException(errorMessage));
        }
    }

    public static void onCompilationProgress(long callbackId, long bytesProcessed, long totalBytes, String phase) {
        CompilationCallback callback = COMPILATION_CALLBACKS.get(callbackId);
        if (callback != null && callback.progress != null) {
            callback.progress.onProgress(bytesProcessed, totalBytes, phase);
        }
    }

    public static void onFunctionCallSuccess(long callbackId, long resultPtr) {
        FunctionCallback callback = FUNCTION_CALLBACKS.remove(callbackId);
        if (callback != null) {
            try {
                WasmValue result = JniValueConverter.fromNative(resultPtr);
                callback.future.complete(result);
            } catch (Exception e) {
                callback.future.completeExceptionally(e);
            }
        }
    }

    public static void onFunctionCallError(long callbackId, String errorMessage) {
        FunctionCallback callback = FUNCTION_CALLBACKS.remove(callbackId);
        if (callback != null) {
            callback.future.completeExceptionally(new WasmException(errorMessage));
        }
    }

    public static void onFunctionCallTimeout(long callbackId) {
        FunctionCallback callback = FUNCTION_CALLBACKS.remove(callbackId);
        if (callback != null) {
            callback.future.completeExceptionally(new WasmTimeoutException("Function call timed out"));
        }
    }

    // Callback data structures
    private static class CompilationCallback {
        final CompletableFuture<Module> future;
        final CompilationProgress progress;

        CompilationCallback(CompletableFuture<Module> future, CompilationProgress progress) {
            this.future = future;
            this.progress = progress;
        }
    }

    private static class FunctionCallback {
        final CompletableFuture<WasmValue> future;
        final Store store;

        FunctionCallback(CompletableFuture<WasmValue> future, Store store) {
            this.future = future;
            this.store = store;
        }
    }

    // Native method declarations
    private native boolean nativeCompileAsync(long enginePtr, byte[] wasmBytes, int length, long callbackId);
    private native boolean nativeCompileInBackground(long enginePtr, byte[] wasmBytes, int length, int priority, long callbackId);
}

public class JniFunction implements WasmFunction {

    // Existing methods...

    @Override
    public CompletableFuture<WasmValue> callAsync(Store store, WasmValue... parameters) {
        return callAsync(store, Duration.ZERO, null, parameters);
    }

    @Override
    public CompletableFuture<WasmValue> callAsync(Store store, Duration timeout, WasmValue... parameters) {
        return callAsync(store, timeout, null, parameters);
    }

    @Override
    public CompletableFuture<WasmValue> callAsync(Store store, CancellationToken cancellationToken, WasmValue... parameters) {
        return callAsync(store, Duration.ZERO, cancellationToken, parameters);
    }

    @Override
    public CompletableFuture<WasmValue> callAsync(Store store, Duration timeout, CancellationToken cancellationToken, WasmValue... parameters) {
        CompletableFuture<WasmValue> future = new CompletableFuture<>();
        long callbackId = ((JniEngine) store.getEngine()).registerFunctionCallback(future, store);

        // Convert parameters to native format
        long[] paramPtrs = new long[parameters.length];
        for (int i = 0; i < parameters.length; i++) {
            paramPtrs[i] = JniValueConverter.toNative(parameters[i]);
        }

        long timeoutMs = timeout.toMillis();
        if (nativeCallAsync(functionPtr, ((JniStore) store).getStorePtr(), paramPtrs, paramPtrs.length, callbackId, timeoutMs)) {
            // Set up cancellation if provided
            if (cancellationToken != null) {
                cancellationToken.onCancellationRequested(() -> {
                    if (!future.isDone()) {
                        nativeCancelCall(callbackId);
                        future.cancel(true);
                    }
                });
            }

            return future;
        } else {
            future.completeExceptionally(new WasmException("Failed to start async function call"));
            return future;
        }
    }

    @Override
    public void callAsync(Store store, WasmValue[] parameters, AsyncCallback<WasmValue> callback) {
        callAsync(store, parameters)
            .whenComplete((result, throwable) -> {
                if (throwable != null) {
                    callback.onError(throwable);
                } else {
                    callback.onSuccess(result);
                }
            });
    }

    @Override
    public CompletableFuture<WasmValue[]> callBatchAsync(Store store, WasmValue[]... parameterSets) {
        CompletableFuture<WasmValue[]> future = new CompletableFuture<>();

        // Convert all parameter sets to native format
        long[][] nativeParamSets = new long[parameterSets.length][];
        int[] paramCounts = new int[parameterSets.length];

        for (int i = 0; i < parameterSets.length; i++) {
            nativeParamSets[i] = new long[parameterSets[i].length];
            paramCounts[i] = parameterSets[i].length;
            for (int j = 0; j < parameterSets[i].length; j++) {
                nativeParamSets[i][j] = JniValueConverter.toNative(parameterSets[i][j]);
            }
        }

        long callbackId = ((JniEngine) store.getEngine()).registerBatchCallback(future, store);

        if (nativeCallBatchAsync(functionPtr, ((JniStore) store).getStorePtr(),
                                nativeParamSets, paramCounts, parameterSets.length, callbackId)) {
            return future;
        } else {
            future.completeExceptionally(new WasmException("Failed to start async batch call"));
            return future;
        }
    }

    // Publisher/Subscriber reactive integration
    @Override
    public Publisher<WasmValue> asPublisher(Store store) {
        return new WasmFunctionPublisher(this, store);
    }

    @Override
    public void executeReactive(Store store, Publisher<WasmValue[]> parameterStream, Subscriber<WasmValue> resultSubscriber) {
        ReactiveWasmFunction reactiveFunction = new ReactiveWasmFunction(this, store);
        reactiveFunction.execute(parameterStream, resultSubscriber);
    }

    // Native method declarations
    private native boolean nativeCallAsync(long functionPtr, long storePtr, long[] paramPtrs, int paramCount, long callbackId, long timeoutMs);
    private native boolean nativeCallBatchAsync(long functionPtr, long storePtr, long[][] paramSets, int[] paramCounts, int batchSize, long callbackId);
    private native boolean nativeCancelCall(long callbackId);
}

// Reactive WebAssembly function implementation
public class ReactiveWasmFunction {
    private final WasmFunction function;
    private final Store store;

    public ReactiveWasmFunction(WasmFunction function, Store store) {
        this.function = function;
        this.store = store;
    }

    public void execute(Publisher<WasmValue[]> parameterStream, Subscriber<WasmValue> resultSubscriber) {
        parameterStream.subscribe(new Subscriber<WasmValue[]>() {
            private Subscription subscription;

            @Override
            public void onSubscribe(Subscription s) {
                this.subscription = s;
                resultSubscriber.onSubscribe(new Subscription() {
                    @Override
                    public void request(long n) {
                        s.request(n);
                    }

                    @Override
                    public void cancel() {
                        s.cancel();
                    }
                });
                s.request(1);
            }

            @Override
            public void onNext(WasmValue[] parameters) {
                function.callAsync(store, parameters)
                    .whenComplete((result, throwable) -> {
                        if (throwable != null) {
                            resultSubscriber.onError(throwable);
                        } else {
                            resultSubscriber.onNext(result);
                            subscription.request(1);
                        }
                    });
            }

            @Override
            public void onError(Throwable t) {
                resultSubscriber.onError(t);
            }

            @Override
            public void onComplete() {
                resultSubscriber.onComplete();
            }
        });
    }
}

        return CompletableFuture.supplyAsync(() -> {
            try (Arena arena = Arena.ofConfined()) {
                MemorySegment wasmBytesSegment = arena.allocateArray(ValueLayout.JAVA_BYTE, wasmBytes);
                MemorySegment resultSegment = (MemorySegment) compileAsyncHandle.invokeExact(
                    engineSegment, wasmBytesSegment, wasmBytes.length
                );

                if (resultSegment.address() == 0) {
                    throw new WasmException("Module compilation failed");
                }

                return new PanamaModule(resultSegment);
            } catch (Throwable e) {
                throw new WasmException("Async compilation failed", e);
            }
        }, ForkJoinPool.commonPool());
    }

    @Override
    public CompletableFuture<Module> compileAsync(byte[] wasmBytes, CompilationProgress progress) {
        return CompletableFuture.supplyAsync(() -> {
            try (Arena arena = Arena.ofConfined()) {
                // Create progress callback structure
                MemorySegment progressCallback = createProgressCallback(arena, progress);
                MemorySegment wasmBytesSegment = arena.allocateArray(ValueLayout.JAVA_BYTE, wasmBytes);

                MemorySegment resultSegment = (MemorySegment) compileAsyncWithProgressHandle.invokeExact(
                    engineSegment, wasmBytesSegment, wasmBytes.length, progressCallback

// Enhanced Panama Function with async operations
public class PanamaFunction implements WasmFunction {

    // Existing methods...

    @Override
    public CompletableFuture<WasmValue> callAsync(Store store, WasmValue... parameters) {
        return CompletableFuture.supplyAsync(() -> {
            try (Arena arena = Arena.ofConfined()) {
                // Convert parameters to native memory segments
                MemorySegment[] paramSegments = new MemorySegment[parameters.length];
                for (int i = 0; i < parameters.length; i++) {
                    paramSegments[i] = PanamaValueConverter.toNative(arena, parameters[i]);
                }

                MemorySegment paramArray = arena.allocateArray(ValueLayout.ADDRESS, paramSegments.length);
                for (int i = 0; i < paramSegments.length; i++) {
                    paramArray.setAtIndex(ValueLayout.ADDRESS, i, paramSegments[i]);
                }

                MemorySegment resultSegment = (MemorySegment) callAsyncHandle.invokeExact(
                    functionSegment,
                    ((PanamaStore) store).getStoreSegment(),
                    paramArray,
                    parameters.length
                );

                if (resultSegment.address() == 0) {
                    throw new WasmException("Async function call failed");
                }

                return PanamaValueConverter.fromNative(resultSegment);
            } catch (Throwable e) {
                throw new WasmException("Async function call failed", e);
            }
        }, ForkJoinPool.commonPool());
    }

    @Override
    public CompletableFuture<WasmValue> callAsync(Store store, Duration timeout, WasmValue... parameters) {
        CompletableFuture<WasmValue> future = callAsync(store, parameters);

        if (!timeout.isZero() && !timeout.isNegative()) {
            return future.orTimeout(timeout.toMillis(), TimeUnit.MILLISECONDS);
        }

        return future;
    }

    @Override
    public CompletableFuture<WasmValue> callAsync(Store store, CancellationToken cancellationToken, WasmValue... parameters) {
        CompletableFuture<WasmValue> future = callAsync(store, parameters);

        if (cancellationToken != null) {
            cancellationToken.onCancellationRequested(() -> future.cancel(true));
        }

        return future;
    }

    @Override
    public CompletableFuture<WasmValue[]> callBatchAsync(Store store, WasmValue[]... parameterSets) {
        return CompletableFuture.supplyAsync(() -> {
            try (Arena arena = Arena.ofConfined()) {
                // Convert batch parameters to native format
                MemorySegment batchSegment = createBatchParametersSegment(arena, parameterSets);

                MemorySegment resultSegment = (MemorySegment) callBatchAsyncHandle.invokeExact(
                    functionSegment,
                    ((PanamaStore) store).getStoreSegment(),
                    batchSegment,
                    parameterSets.length
                );

                if (resultSegment.address() == 0) {
                    throw new WasmException("Async batch call failed");
                }

                return convertBatchResults(resultSegment, parameterSets.length);
            } catch (Throwable e) {
                throw new WasmException("Async batch call failed", e);
            }
        }, ForkJoinPool.commonPool());
    }

    // Method handle declarations
    private static final MethodHandle callAsyncHandle;
    private static final MethodHandle callBatchAsyncHandle;
    private static final MethodHandle callCancelHandle;

    static {
        try {
            SymbolLookup lookup = SymbolLookup.loaderLookup();

            callAsyncHandle = Linker.nativeLinker().downcallHandle(
                lookup.find("wasmtime4j_func_call_async").orElseThrow(),
                FunctionDescriptor.of(ValueLayout.ADDRESS, ValueLayout.ADDRESS, ValueLayout.ADDRESS, ValueLayout.ADDRESS, ValueLayout.JAVA_INT)
            );

            callBatchAsyncHandle = Linker.nativeLinker().downcallHandle(
                lookup.find("wasmtime4j_func_call_batch_async").orElseThrow(),
                FunctionDescriptor.of(ValueLayout.ADDRESS, ValueLayout.ADDRESS, ValueLayout.ADDRESS, ValueLayout.ADDRESS, ValueLayout.JAVA_INT)
            );

            callCancelHandle = Linker.nativeLinker().downcallHandle(
                lookup.find("wasmtime4j_func_call_cancel").orElseThrow(),
                FunctionDescriptor.of(ValueLayout.JAVA_BOOLEAN, ValueLayout.JAVA_LONG)
            );

        } catch (Throwable e) {
            throw new RuntimeException("Failed to initialize Panama async function handles", e);
        }
    }

    private MemorySegment createBatchParametersSegment(Arena arena, WasmValue[][] parameterSets) {
        // Implementation for creating native batch parameter structure
        MemorySegment batchSegment = arena.allocate(ValueLayout.ADDRESS, parameterSets.length);

        for (int i = 0; i < parameterSets.length; i++) {
            MemorySegment paramSetSegment = arena.allocateArray(ValueLayout.ADDRESS, parameterSets[i].length);
            for (int j = 0; j < parameterSets[i].length; j++) {
                MemorySegment paramSegment = PanamaValueConverter.toNative(arena, parameterSets[i][j]);
                paramSetSegment.setAtIndex(ValueLayout.ADDRESS, j, paramSegment);
            }
            batchSegment.setAtIndex(ValueLayout.ADDRESS, i, paramSetSegment);
        }

        return batchSegment;
    }

    private WasmValue[] convertBatchResults(MemorySegment resultSegment, int resultCount) {
        WasmValue[] results = new WasmValue[resultCount];

        for (int i = 0; i < resultCount; i++) {
            MemorySegment resultItemSegment = resultSegment.getAtIndex(ValueLayout.ADDRESS, i);
            results[i] = PanamaValueConverter.fromNative(resultItemSegment);
        }

        return results;
    }
}

// Input stream implementation for WebAssembly memory
public class WasmMemoryInputStream extends InputStream {
    private final WasmMemory memory;
    private final Store store;
    private final long startOffset;
    private final long length;
    private long currentOffset;

    public WasmMemoryInputStream(WasmMemory memory, Store store, long offset, long length) {
        this.memory = memory;
        this.store = store;
        this.startOffset = offset;
        this.length = length;
        this.currentOffset = 0;
    }

    @Override
    public int read() throws IOException {
        if (currentOffset >= length) {
            return -1;
        }

        byte[] buffer = new byte[1];
        int bytesRead = read(buffer, 0, 1);
        return bytesRead > 0 ? (buffer[0] & 0xFF) : -1;
    }

    @Override
    public int read(byte[] b, int off, int len) throws IOException {
        if (currentOffset >= length) {
            return -1;
        }

        int bytesToRead = (int) Math.min(len, length - currentOffset);
        memory.read(store, startOffset + currentOffset, b, off, bytesToRead);
        currentOffset += bytesToRead;
        return bytesToRead;
    }

    @Override
    public long skip(long n) throws IOException {
        long bytesToSkip = Math.min(n, length - currentOffset);
        currentOffset += bytesToSkip;
        return bytesToSkip;
    }

    @Override
    public int available() throws IOException {
        return (int) (length - currentOffset);
    }
}

public class WasmMemoryOutputStream extends OutputStream {

    private final Store store;
    private final long startOffset;
    private final long maxLength;
    private long currentOffset;

    public WasmMemoryOutputStream(WasmMemory memory, Store store, long offset, long maxLength) {
        this.memory = memory;
        this.store = store;
        this.startOffset = offset;
        this.maxLength = maxLength;
        this.currentOffset = 0;
    }

    @Override
    public void write(int b) throws IOException {
        write(new byte[]{(byte) b}, 0, 1);
    }

    @Override
    public void write(byte[] b, int off, int len) throws IOException {
        if (currentOffset + len > maxLength) {
            throw new IOException("Write would exceed maximum memory stream length");
        }

        memory.write(store, startOffset + currentOffset, b, off, len);
        currentOffset += len;
    }

    @Override
    public void flush() throws IOException {
        // WebAssembly memory writes are immediately visible
    }
}

// Concrete cancellation token implementations
public class SimpleCancellationToken implements CancellationToken {
    private final AtomicBoolean cancelled = new AtomicBoolean(false);
    private final List<Runnable> callbacks = new CopyOnWriteArrayList<>();
    private volatile String reason;

    @Override
    public boolean isCancellationRequested() {
        return cancelled.get();
    }

    @Override
    public void cancel() {
        cancel("Cancellation requested");
    }

    @Override
    public void cancel(String reason) {
        if (cancelled.compareAndSet(false, true)) {
            this.reason = reason;
            for (Runnable callback : callbacks) {
                try {
                    callback.run();
                } catch (Exception e) {
                    // Log but don't propagate callback exceptions
                }
            }
        }
    }

    @Override
    public void onCancellationRequested(Runnable callback) {
        if (cancelled.get()) {
            callback.run();
        } else {
            callbacks.add(callback);
            // Check again in case cancellation happened between the first check and adding the callback
            if (cancelled.get()) {
                callback.run();
            }
        }
    }

    public String getReason() {
        return reason;
    }
}

public class TimeoutCancellationToken implements CancellationToken {
    private final SimpleCancellationToken delegate = new SimpleCancellationToken();
    private final ScheduledExecutorService scheduler = Executors.newScheduledThreadPool(1);
    private final ScheduledFuture<?> timeoutTask;

    public TimeoutCancellationToken(Duration timeout) {
        this.timeoutTask = scheduler.schedule(
            () -> delegate.cancel("Operation timed out after " + timeout),
            timeout.toMillis(),
            TimeUnit.MILLISECONDS
        );
    }

    @Override
    public boolean isCancellationRequested() {
        return delegate.isCancellationRequested();
    }

    @Override
    public void cancel() {
        cancel("Manual cancellation");
    }

    @Override
    public void cancel(String reason) {
        timeoutTask.cancel(false);
        delegate.cancel(reason);
    }

    @Override
    public void onCancellationRequested(Runnable callback) {
        delegate.onCancellationRequested(callback);
    }

    public void dispose() {
        timeoutTask.cancel(false);
        scheduler.shutdown();
    }
}

public class CombinedCancellationToken implements CancellationToken {
    private final CancellationToken[] tokens;
    private final SimpleCancellationToken combined = new SimpleCancellationToken();

    public CombinedCancellationToken(CancellationToken... tokens) {
        this.tokens = tokens.clone();

        for (CancellationToken token : tokens) {
            token.onCancellationRequested(() -> {
                if (!combined.isCancellationRequested()) {
                    combined.cancel("Combined cancellation triggered");
                }
            });
        }
    }

    @Override
    public boolean isCancellationRequested() {
        return combined.isCancellationRequested() ||
               Arrays.stream(tokens).anyMatch(CancellationToken::isCancellationRequested);
    }

    @Override
    public void cancel() {
        combined.cancel();
    }

    @Override
    public void cancel(String reason) {
        combined.cancel(reason);
    }

    @Override
    public void onCancellationRequested(Runnable callback) {
        combined.onCancellationRequested(callback);
    }
}
