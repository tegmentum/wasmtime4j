// Async runtime infrastructure in wasmtime4j-native
use tokio::runtime::{Builder, Runtime};
use std::sync::Arc;
use std::sync::OnceLock;

// Global async runtime for wasmtime4j
static ASYNC_RUNTIME: OnceLock<Arc<Runtime>> = OnceLock::new();

pub fn get_async_runtime() -> &'static Arc<Runtime> {
    ASYNC_RUNTIME.get_or_init(|| {
        Arc::new(
            Builder::new_multi_thread()
                .worker_threads(4)
                .thread_name("wasmtime4j-async")
                .enable_all()
                .build()
                .expect("Failed to create async runtime")
        )
    })
}

// Async function call implementation
#[no_mangle]
pub extern "C" fn wasmtime4j_func_call_async(
    func_ptr: *mut c_void,
    store_ptr: *mut c_void,
    params_ptr: *const *const c_void,
    params_count: u32,
    callback_id: u64,
    timeout_ms: u64
) -> bool {
    let func = unsafe { &*(func_ptr as *const wasmtime::Func) };
    let mut store = unsafe { &mut *(store_ptr as *mut wasmtime::Store<()>) };

    // Convert parameters
    let params = unsafe {
        std::slice::from_raw_parts(params_ptr, params_count as usize)
            .iter()
            .map(|&ptr| convert_java_to_wasmtime_val(ptr))
            .collect::<Vec<_>>()
    };

    let rt = get_async_runtime();
    rt.spawn(async move {
        let future = async {
            let mut results = vec![wasmtime::Val::I32(0); func.ty(&store).results().len()];
            func.call_async(&mut store, &params, &mut results).await
        };

        let result = if timeout_ms > 0 {
            tokio::time::timeout(
                std::time::Duration::from_millis(timeout_ms),
                future
            ).await
        } else {
            Ok(future.await)
        };

        match result {
            Ok(Ok(results)) => {
                notify_async_success(callback_id, results);
            }
            Ok(Err(error)) => {
                notify_async_error(callback_id, format!("Call failed: {}", error));
            }
            Err(_) => {
                notify_async_timeout(callback_id);
            }
        }
    });

    true
}

pub extern "C" fn wasmtime4j_module_compile_async(
    engine_ptr: *mut c_void,
    wasm_bytes: *const u8,
    bytes_len: u32,
    callback_id: u64
) -> bool {
    let engine = unsafe { &*(engine_ptr as *const wasmtime::Engine) };
    let bytes = unsafe { std::slice::from_raw_parts(wasm_bytes, bytes_len as usize) }.to_vec();

    let rt = get_async_runtime();
    rt.spawn_blocking(move || {
        match wasmtime::Module::new(&engine, &bytes) {
            Ok(module) => {
                let module_ptr = Box::into_raw(Box::new(module)) as *mut c_void;
                notify_compilation_success(callback_id, module_ptr);
            }
            Err(error) => {
                notify_compilation_error(callback_id, format!("Compilation failed: {}", error));
            }
        }
    });

    true
}

    instance_ptr: *mut c_void,
    store_ptr: *mut c_void,

    function_name: *const c_char,
    params_ptr: *const *const c_void,
    params_count: u32,
    results_ptr: *mut *mut c_void
) -> bool {

    let mut store = unsafe { &mut *(store_ptr as *mut wasmtime::Store<()>) };
    let interface_name = unsafe { CStr::from_ptr(interface_name).to_str().unwrap() };
    let function_name = unsafe { CStr::from_ptr(function_name).to_str().unwrap() };

    // Get the exported function
    let func = match instance.get_func(&mut store, &format!("{}#{}", interface_name, function_name)) {
        Some(func) => func,
        None => return false,
    };

    // Convert parameters
    let params = unsafe {
        std::slice::from_raw_parts(params_ptr, params_count as usize)
            .iter()
            .map(|&ptr| convert_java_to_component_val(ptr))
            .collect::<Vec<_>>()

    let mut results = vec![Val::Bool(false); func.ty(&store).results().len()];

        Ok(_) => {
            // Convert results back to Java

                unsafe {
                    *results_ptr.add(i) = convert_component_val_to_java(result);
                }
            }
            true
        }
        Err(_) => false,
    }
}

pub extern "C" fn wasmtime4j_memory_bulk_copy(
    memory_ptr: *mut c_void,
    store_ptr: *mut c_void,
    dest_offset: u64,
    src_offset: u64,
    length: u64
) -> bool {
    let memory = unsafe { &*(memory_ptr as *const wasmtime::Memory) };
    let mut store = unsafe { &mut *(store_ptr as *mut wasmtime::Store<()>) };

    // Use Wasmtime's efficient bulk memory operations
    match memory.data(&store).get(src_offset as usize..(src_offset + length) as usize) {
        Some(src_data) => {
            let data_copy = src_data.to_vec();
            match memory.write(&mut store, dest_offset as usize, &data_copy) {
                Ok(_) => true,
                Err(_) => false,
            }
        }
        None => false,
    }
}

// Performance monitoring infrastructure
use std::time::{Instant, Duration};
use std::collections::HashMap;
use std::sync::{Mutex, Arc};

#[derive(Clone)]
pub struct PerformanceTracker {
    start_time: Instant,
    function_calls: Arc<Mutex<HashMap<String, CallStatistics>>>,
    compilation_stats: Arc<Mutex<CompilationStats>>,
    memory_stats: Arc<Mutex<MemoryStats>>,
}

#[derive(Clone)]
struct CallStatistics {
    count: u64,
    total_time: Duration,
    min_time: Duration,
    max_time: Duration,
}

#[derive(Clone)]
struct CompilationStats {
    modules_compiled: u64,
    total_compilation_time: Duration,
    cache_hits: u64,
    cache_misses: u64,
}

#[derive(Clone)]
struct MemoryStats {
    total_allocated: u64,
    total_deallocated: u64,
    peak_usage: u64,
    current_usage: u64,
}

static PERFORMANCE_TRACKER: OnceLock<PerformanceTracker> = OnceLock::new();

#[no_mangle]
pub extern "C" fn wasmtime4j_perf_init() -> bool {
    PERFORMANCE_TRACKER.set(PerformanceTracker {
        start_time: Instant::now(),
        function_calls: Arc::new(Mutex::new(HashMap::new())),
        compilation_stats: Arc::new(Mutex::new(CompilationStats {
            modules_compiled: 0,
            total_compilation_time: Duration::ZERO,
            cache_hits: 0,
            cache_misses: 0,
        })),
        memory_stats: Arc::new(Mutex::new(MemoryStats {
            total_allocated: 0,
            total_deallocated: 0,
            peak_usage: 0,
            current_usage: 0,
        })),
    }).is_ok()
}

#[no_mangle]
pub extern "C" fn wasmtime4j_perf_record_function_call(
    function_name: *const c_char,
    execution_time_ns: u64
) {
    if let Some(tracker) = PERFORMANCE_TRACKER.get() {
        let name = unsafe { CStr::from_ptr(function_name).to_str().unwrap_or("unknown") };
        let duration = Duration::from_nanos(execution_time_ns);

        if let Ok(mut calls) = tracker.function_calls.lock() {
            let stats = calls.entry(name.to_string()).or_insert(CallStatistics {
                count: 0,
                total_time: Duration::ZERO,
                min_time: Duration::MAX,
                max_time: Duration::ZERO,
            });

            stats.count += 1;
            stats.total_time += duration;
            stats.min_time = stats.min_time.min(duration);
            stats.max_time = stats.max_time.max(duration);
        }
    }
}

#[no_mangle]
pub extern "C" fn wasmtime4j_perf_get_engine_statistics(
    stats_ptr: *mut EngineStatsNative
) -> bool {
    if let Some(tracker) = PERFORMANCE_TRACKER.get() {
        let stats = unsafe { &mut *stats_ptr };

        stats.uptime_ms = tracker.start_time.elapsed().as_millis() as u64;

        if let Ok(calls) = tracker.function_calls.lock() {
            stats.total_function_calls = calls.values().map(|s| s.count).sum();
            stats.average_call_time_ns = if calls.is_empty() {
                0
            } else {
                calls.values().map(|s| s.total_time.as_nanos() as u64 / s.count).sum::<u64>() / calls.len() as u64
            };
        }

        if let Ok(comp_stats) = tracker.compilation_stats.lock() {
            stats.modules_compiled = comp_stats.modules_compiled;
            stats.total_compilation_time_ms = comp_stats.total_compilation_time.as_millis() as u64;
        }

        if let Ok(mem_stats) = tracker.memory_stats.lock() {
            stats.current_memory_usage = mem_stats.current_usage;
            stats.peak_memory_usage = mem_stats.peak_usage;
        }

        true
    } else {
        false
    }
}

#[repr(C)]
pub struct EngineStatsNative {
    pub uptime_ms: u64,
    pub total_function_calls: u64,
    pub average_call_time_ns: u64,
    pub modules_compiled: u64,
    pub total_compilation_time_ms: u64,
    pub current_memory_usage: u64,
    pub peak_memory_usage: u64,
}
