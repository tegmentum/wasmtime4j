# Task: Comprehensive Cross-Platform Testing

## Objective

Implement comprehensive cross-platform testing infrastructure to validate 100% API coverage across all supported platforms (Linux, Windows, macOS) and architectures (x86_64, ARM64). Ensure identical behavior between JNI and Panama implementations while validating performance and reliability requirements.

```java
// src/test/java/ai/tegmentum/wasmtime4j/CrossPlatformTestSuite.java
@TestMethodOrder(OrderAnnotation.class)
public class CrossPlatformTestSuite {

    @TestFactory
    @Order(1)
    Stream<DynamicTest> platformCompatibilityTests() {
        return Platform.getSupportedPlatforms().stream()
            .map(platform -> DynamicTest.dynamicTest(
                "Platform compatibility: " + platform.getName(),
                () -> validatePlatformCompatibility(platform)
            ));
    }

    @TestFactory
    @Order(2)
    Stream<DynamicTest> runtimeImplementationTests() {
        return RuntimeType.values().stream()
            .filter(RuntimeType::isAvailable)
            .map(runtime -> DynamicTest.dynamicTest(
                "Runtime implementation: " + runtime.name(),
                () -> validateRuntimeImplementation(runtime)
            ));
    }

    @TestFactory
    @Order(3)
    Stream<DynamicTest> crossRuntimeConsistencyTests() {
        List<RuntimeType> availableRuntimes = Arrays.stream(RuntimeType.values())
            .filter(RuntimeType::isAvailable)
            .collect(Collectors.toList());

        if (availableRuntimes.size() < 2) {
            return Stream.of(DynamicTest.dynamicTest(
                "Cross-runtime consistency: SKIPPED (only one runtime available)",
                () -> Assumptions.assumeTrue(false, "Need multiple runtimes for consistency testing")
            ));
        }

        return generateCrossRuntimeTests(availableRuntimes);
    }

    private void validatePlatformCompatibility(Platform platform) {
        // Test native library loading
        assertDoesNotThrow(() -> NativeLibraryLoader.loadForPlatform(platform));

        // Test basic engine creation
        Config config = Config.builder().build();
        try (Engine engine = Engine.create(config)) {
            assertNotNull(engine);
            assertNotNull(engine.getConfig());
        }

        // Test WebAssembly module compilation
        byte[] wasmBytes = TestResources.getSimpleWasmModule();
        try (Engine engine = Engine.create(config);
             Module module = Module.compile(engine, wasmBytes)) {
            assertNotNull(module);
            assertTrue(module.getExports().size() > 0);
        }
    }

    private Stream<DynamicTest> generateCrossRuntimeTests(List<RuntimeType> runtimes) {
        return TestScenarios.getAllScenarios().stream()
            .flatMap(scenario -> runtimes.stream()
                .flatMap(runtime1 -> runtimes.stream()
                    .filter(runtime2 -> !runtime1.equals(runtime2))
                    .map(runtime2 -> DynamicTest.dynamicTest(
                        String.format("Consistency: %s vs %s - %s",
                            runtime1.name(), runtime2.name(), scenario.getName()),
                        () -> validateConsistency(runtime1, runtime2, scenario)
                    ))
                )
            );
    }
}
```

```java
// src/test/java/ai/tegmentum/wasmtime4j/performance/PerformanceValidationSuite.java
@BenchmarkMode(Mode.AverageTime)
@OutputTimeUnit(TimeUnit.MICROSECONDS)
@State(Scope.Benchmark)
public class PerformanceValidationSuite {

    @Param({"JNI", "PANAMA"})
    private String runtimeType;

    @Param({"simple", "complex", "memory_intensive", "compute_intensive"})
    private String testScenario;

    private Engine engine;
    private Module module;
    private Store store;
    private Instance instance;

    @Setup(Level.Trial)
    public void setupTrial() {
        System.setProperty("wasmtime4j.runtime", runtimeType.toLowerCase());

        Config config = Config.builder()
            .withOptimizationLevel(OptimizationLevel.SPEED)
            .build();

        engine = Engine.create(config);
        byte[] wasmBytes = TestResources.getModuleByName(testScenario);
        module = Module.compile(engine, wasmBytes);
        store = Store.create(engine);
        instance = Instance.create(store, module, new ArrayList<>());
    }

    @TearDown(Level.Trial)
    public void teardownTrial() {
        if (instance != null) instance.close();
        if (store != null) store.close();
        if (module != null) module.close();
        if (engine != null) engine.close();
    }

    @Benchmark
    public Object functionCall() throws Exception {
        WasmFunction function = instance.getExportedFunction("test_function")
            .orElseThrow(() -> new IllegalStateException("Function not found"));

        return function.call(store, 42, 3.14f);
    }

    @Benchmark
    public void memoryAccess() {
        Memory memory = instance.getExportedMemory("memory")
            .orElseThrow(() -> new IllegalStateException("Memory not found"));

        // Test bulk memory operations
        byte[] data = new byte[1024];
        Arrays.fill(data, (byte) 0xFF);
        memory.write(store, 0, data);

        byte[] readData = new byte[1024];
        memory.read(store, 0, readData);
    }

    @Benchmark
    public Instance instantiation() {
        try (Instance inst = Instance.create(store, module, new ArrayList<>())) {
            return inst;
        }
    }

    // Performance requirements validation
    @Test
    public void validatePerformanceRequirements() {
        Map<String, Double> benchmarkResults = runBenchmarks();

        // Host function binding within 10% of native performance
        double hostFunctionOverhead = benchmarkResults.get("host_function_overhead");
        assertTrue(hostFunctionOverhead < 0.10,
            "Host function overhead exceeds 10%: " + hostFunctionOverhead);

        // Type introspection minimal memory overhead (<5% increase)
        double memoryOverhead = benchmarkResults.get("type_introspection_memory_overhead");
        assertTrue(memoryOverhead < 0.05,
            "Type introspection memory overhead exceeds 5%: " + memoryOverhead);

        // Configuration access no impact on engine creation
        double configAccessImpact = benchmarkResults.get("config_access_impact");
        assertTrue(configAccessImpact < 0.01,
            "Configuration access impacts engine creation: " + configAccessImpact);
    }
}
```

### API Coverage Validation

Implement comprehensive API coverage testing:

```java
// src/test/java/ai/tegmentum/wasmtime4j/coverage/APICoverageValidationSuite.java
public class APICoverageValidationSuite {

    @TestFactory
    Stream<DynamicTest> engineAPICoverage() {
        return createAPICoverageTests(Engine.class, this::testEngineAPI);
    }

    @TestFactory
    Stream<DynamicTest> linkerAPICoverage() {
        return createAPICoverageTests(Linker.class, this::testLinkerAPI);
    }

    @TestFactory
    Stream<DynamicTest> typeIntrospectionAPICoverage() {
        return Stream.of(
            MemoryType.class, TableType.class, GlobalType.class, FuncType.class
        ).flatMap(typeClass -> createAPICoverageTests(typeClass, this::testTypeAPI));
    }

    @TestFactory
    Stream<DynamicTest> externAPICoverage() {
        return Stream.of(
            ExternFunction.class, ExternMemory.class, ExternTable.class, ExternGlobal.class
        ).flatMap(externClass -> createAPICoverageTests(externClass, this::testExternAPI));
    }

    private <T> Stream<DynamicTest> createAPICoverageTests(Class<T> apiClass, Consumer<Method> methodTester) {
        return Arrays.stream(apiClass.getMethods())
            .filter(method -> !method.getDeclaringClass().equals(Object.class))
            .filter(method -> !method.isDefault())
            .map(method -> DynamicTest.dynamicTest(
                String.format("API Coverage: %s.%s", apiClass.getSimpleName(), method.getName()),
                () -> methodTester.accept(method)
            ));
    }

    private void testEngineAPI(Method method) {
        Config config = Config.builder().build();
        try (Engine engine = Engine.create(config)) {
            validateMethodImplementation(engine, method);
        }
    }

    private void testLinkerAPI(Method method) {
        Config config = Config.builder().build();
        try (Engine engine = Engine.create(config);
             Linker linker = Linker.create(engine)) {
            validateMethodImplementation(linker, method);
        }
    }

    private void validateMethodImplementation(Object instance, Method method) {
        // Verify method is not a stub implementation
        assertFalse(isStubImplementation(instance, method),
            "Method " + method.getName() + " appears to be a stub implementation");

        // Verify method behaves consistently across runtimes
        if (RuntimeType.values().length > 1) {
            validateCrossRuntimeConsistency(instance.getClass(), method);
        }
    }

    @Test
    public void validate100PercentAPICoverage() {
        List<String> missingAPIs = findMissingAPIs();
        assertTrue(missingAPIs.isEmpty(),
            "Missing API implementations: " + String.join(", ", missingAPIs));
    }

    private List<String> findMissingAPIs() {
        List<String> missing = new ArrayList<>();

        // Check against Wasmtime API reference
        WasmtimeAPIReference reference = WasmtimeAPIReference.load();
        for (String apiMethod : reference.getAllMethods()) {
            if (!isAPIImplemented(apiMethod)) {
                missing.add(apiMethod);
            }
        }

        return missing;
    }
}
```

### Error Handling Validation

Comprehensive error handling testing:

```java
// src/test/java/ai/tegmentum/wasmtime4j/error/ErrorHandlingValidationSuite.java
public class ErrorHandlingValidationSuite {

    @TestFactory
    Stream<DynamicTest> errorHandlingConsistency() {
        return ErrorScenario.getAllScenarios().stream()
            .map(scenario -> DynamicTest.dynamicTest(
                "Error handling: " + scenario.getName(),
                () -> validateErrorHandling(scenario)
            ));
    }

    private void validateErrorHandling(ErrorScenario scenario) {
        Map<RuntimeType, Exception> runtimeExceptions = new HashMap<>();

        for (RuntimeType runtime : RuntimeType.values()) {
            if (!runtime.isAvailable()) continue;

            System.setProperty("wasmtime4j.runtime", runtime.name().toLowerCase());

            Exception exception = assertThrows(Exception.class, scenario::execute);
            runtimeExceptions.put(runtime, exception);
        }

        // Validate consistent error types across runtimes
        validateErrorConsistency(runtimeExceptions);
    }

    private void validateErrorConsistency(Map<RuntimeType, Exception> exceptions) {
        if (exceptions.size() < 2) return;

        List<Exception> exceptionList = new ArrayList<>(exceptions.values());
        Exception first = exceptionList.get(0);

        for (int i = 1; i < exceptionList.size(); i++) {
            Exception current = exceptionList.get(i);

            // Same exception type
            assertEquals(first.getClass(), current.getClass(),
                "Exception types differ between runtimes");

            // Similar error messages
            assertTrue(areErrorMessagesSimilar(first.getMessage(), current.getMessage()),
                "Error messages differ significantly between runtimes");
        }
    }

    @Test
    public void validateNoJVMCrashes() {
        // Test various error conditions that should not crash the JVM
        List<Runnable> crashTests = Arrays.asList(
            this::testInvalidModuleBytes,
            this::testMemoryOutOfBounds,
            this::testInvalidFunctionSignature,
            this::testResourceExhaustion,
            this::testConcurrentAccess
        );

        for (Runnable test : crashTests) {
            assertDoesNotThrow(test::run, "JVM crash detected during error condition test");
        }
    }
}
```

### Memory Management Validation

Comprehensive memory leak detection:

```java
// src/test/java/ai/tegmentum/wasmtime4j/memory/MemoryManagementValidationSuite.java
public class MemoryManagementValidationSuite {

    @Test
    public void validateNoMemoryLeaks() {
        long initialMemory = getUsedMemory();

        // Perform intensive operations
        for (int i = 0; i < 1000; i++) {
            performCompleteWorkflow();

            if (i % 100 == 0) {
                System.gc();
                Thread.sleep(100);
            }
        }

        System.gc();
        Thread.sleep(1000);

        long finalMemory = getUsedMemory();
        long memoryIncrease = finalMemory - initialMemory;

        assertTrue(memoryIncrease < 50_000_000, // 50MB threshold
            "Memory leak detected: " + memoryIncrease + " bytes increased");
    }

    private void performCompleteWorkflow() {
        Config config = Config.builder()
            .withOptimizationLevel(OptimizationLevel.SPEED)
            .build();

        try (Engine engine = Engine.create(config);
             Linker linker = Linker.create(engine)) {

            byte[] wasmBytes = TestResources.getComplexModule();
            Module module = Module.compile(engine, wasmBytes);

            Store store = Store.create(engine);
            Instance instance = Instance.create(store, module, new ArrayList<>());

            // Exercise all APIs
            exerciseAllAPIs(engine, linker, module, store, instance);

            instance.close();
            store.close();
            module.close();
        }
    }

    @Test
    public void validateResourceCleanup() {
        ResourceTracker tracker = new ResourceTracker();

        try {
            performResourceIntensiveOperations(tracker);
        } finally {
            tracker.validateAllResourcesCleaned();
        }
    }
}
```
