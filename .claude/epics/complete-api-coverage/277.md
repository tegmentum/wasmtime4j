// wasmtime4j-native/src/wasi_preview2.rs
use wasmtime::component::{Component, Linker, ResourceTable};
use wasmtime_wasi::{WasiView, WasiCtx};

pub struct WasiP2Context {
    wasi_ctx: WasiCtx,
    resource_table: ResourceTable,
    component_linker: Option<Linker<WasiP2ContextData>>,
    interfaces: HashMap<String, Box<dyn WasiInterface>>,
}

pub struct WasiP2ContextData {
    wasi_ctx: WasiCtx,
    resource_table: ResourceTable,
}

impl WasiView for WasiP2ContextData {
    fn ctx(&self) -> &WasiCtx { &self.wasi_ctx }
    fn ctx_mut(&mut self) -> &mut WasiCtx { &mut self.wasi_ctx }
    fn table(&self) -> &ResourceTable { &self.resource_table }
    fn table_mut(&mut self) -> &mut ResourceTable { &mut self.resource_table }
}

impl WasiP2Context {
    pub fn new() -> WasmtimeResult<Self> {
        let wasi_ctx = WasiCtxBuilder::new().build();
        let resource_table = ResourceTable::new();

        Ok(WasiP2Context {
            wasi_ctx,
            resource_table,
            component_linker: None,
            interfaces: HashMap::new(),
        })
    }

    pub fn create_resource<T>(&mut self, resource: T) -> WasmtimeResult<u32>
    where
        T: Send + 'static
    {
        let handle = self.resource_table.push(resource)?;
        Ok(handle.rep())
    }

    pub fn get_resource<T>(&self, handle: u32) -> WasmtimeResult<&T>
    where
        T: Send + 'static
    {
        let resource_handle = wasmtime::component::ResourceAny::from_rep(handle);
        self.resource_table.get(&resource_handle)
    }

    pub fn drop_resource(&mut self, handle: u32) -> WasmtimeResult<()> {
        let resource_handle = wasmtime::component::ResourceAny::from_rep(handle);
        self.resource_table.delete(resource_handle)?;
        Ok(())
    }

    // Interface management
    pub fn bind_interface(&mut self, name: &str, interface: Box<dyn WasiInterface>) -> WasmtimeResult<()> {
        self.interfaces.insert(name.to_string(), interface);
        Ok(())
    }

    pub fn get_interface(&self, name: &str) -> Option<&dyn WasiInterface> {
        self.interfaces.get(name).map(|i| i.as_ref())
    }
}

// Component interface trait
pub trait WasiInterface: Send + Sync {
    fn get_exports(&self) -> Vec<WasiExport>;
    fn call_function(&self, name: &str, args: &[WasiValue]) -> WasmtimeResult<Vec<WasiValue>>;
}
