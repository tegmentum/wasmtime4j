// Example of comprehensive Javadoc for Linker interface
/**
 * A {@code Linker} provides a way to define host functions and link WebAssembly modules
 * with their imports satisfied. The linker enables dynamic linking of WebAssembly modules
 * by providing a registry of available imports and managing the instantiation process.
 *
 * <h2>Basic Usage</h2>
 * <pre>{@code
 * Config config = Config.builder().build();
 * try (Engine engine = Engine.create(config);
 *      Linker linker = Linker.create(engine)) {
 *
 *     // Define a host function
 *     linker.defineHostFunction("env", "log", (store, params) -> {
 *         System.out.println("Log: " + params[0].asInt());
 *         return new Object[0];
 *     });
 *
 *     // Compile and instantiate a module
 *     Module module = Module.compile(engine, wasmBytes);
 *     Store store = Store.create(engine);
 *     Instance instance = linker.instantiate(store, module);
 * }
 * }</pre>
 *
 * <h2>Host Function Definition</h2>
 * <p>Host functions can be defined using several approaches:</p>
 * <ul>
 *   <li>{@link #defineHostFunction(String, String, HostFunction)} - Define a function with callback</li>
 *   <li>{@link #define(String, String, WasmFunction)} - Define using an existing WebAssembly function</li>
 *   <li>{@link #defineExtern(String, String, Extern)} - Define using an external object</li>
 * </ul>
 *
 * <h2>Module Composition</h2>
 * <p>The linker supports sophisticated module composition patterns:</p>
 * <pre>{@code
 * // Link multiple modules together
 * linker.linkModule("math", mathModule);
 * linker.linkInstance("io", ioInstance);
 *
 * // Create aliases between modules
 * linker.alias("app", "sqrt", "math", "sqrt");
 * }</pre>
 *
 * <h2>WASI Integration</h2>
 * <p>WASI (WebAssembly System Interface) can be easily integrated:</p>
 * <pre>{@code
 * WasiConfig wasiConfig = WasiConfig.builder()
 *     .withStandardStreams()
 *     .withEnvironmentVariable("HOME", "/tmp")
 *     .build();
 * linker.defineWasi(wasiConfig);
 * }</pre>
 *
 * <h2>Thread Safety</h2>
 * <p>Linker instances are <strong>not</strong> thread-safe. Each thread should use its own
 * linker instance, or external synchronization must be provided when sharing a linker
 * between threads.</p>
 *
 * <h2>Resource Management</h2>
 * <p>Linkers implement {@link AutoCloseable} and should be used in try-with-resources
 * blocks to ensure proper cleanup of native resources.</p>
 *
 * <h2>Performance Considerations</h2>
 * <p>For optimal performance:</p>
 * <ul>
 *   <li>Reuse linker instances when possible</li>
 *   <li>Pre-define commonly used host functions</li>
 *   <li>Use {@link #resolveImport(String, String)} to validate imports before instantiation</li>
 * </ul>
 *
 * @since 1.0.0
 * @see Engine
 * @see Module
 * @see Instance
 * @see HostFunction
 */
public interface Linker extends AutoCloseable {

    /**
     * Defines a host function that can be imported by WebAssembly modules.
     *
     * <p>The host function will be available to WebAssembly modules that import
     * a function with the specified module name and function name. The function
     * signature must match the import declaration in the WebAssembly module.</p>
     *
     * <h3>Example</h3>
     * <pre>{@code
     * // Define a simple logging function
     * linker.defineHostFunction("env", "log", (store, params) -> {
     *     int value = params[0].asInt();
     *     System.out.println("WebAssembly logged: " + value);
     *     return new Object[0]; // No return values
     * });
     *
     * // Define a function that returns a value
     * linker.defineHostFunction("env", "get_time", (store, params) -> {
     *     long currentTime = System.currentTimeMillis();
     *     return new Object[]{currentTime};
     * });
     * }</pre>
     *
     * <h3>Parameter and Return Value Handling</h3>
     * <p>Parameters are provided as an array of {@link Val} objects. Use the appropriate
     * conversion methods like {@link Val#asInt()}, {@link Val#asLong()}, {@link Val#asFloat()},
     * etc. to extract typed values.</p>
     *
     * <p>Return values should be provided as an Object array where each element
     * corresponds to a return value. The types will be automatically converted
     * to the appropriate WebAssembly types based on the function signature.</p>
     *
     * @param module the module name for the import (e.g., "env", "wasi_snapshot_preview1")
     * @param name the function name for the import (e.g., "log", "fd_write")
     * @param function the host function implementation
     *
     * @throws ValidationException if a function with the same module and name is already defined
     * @throws IllegalArgumentException if any parameter is null or empty
     * @throws IllegalStateException if this linker has been closed
     *
     * @see HostFunction
     * @see Val
     */
    void defineHostFunction(String module, String name, HostFunction function);

}

// src/examples/java/ai/tegmentum/wasmtime4j/examples/LinkerExamples.java
/**
 * Comprehensive examples demonstrating Linker usage patterns.
 */
public class LinkerExamples {

    /**
     * Basic host function definition and usage.
     */
    public static void basicHostFunctionExample() {
        Config config = Config.builder().build();

        try (Engine engine = Engine.create(config);
             Linker linker = Linker.create(engine)) {

            // Define a simple host function
            linker.defineHostFunction("env", "print", (store, params) -> {
                System.out.println("From WASM: " + params[0].asInt());
                return new Object[0];
            });

            // Define a function that returns a value
            linker.defineHostFunction("env", "get_random", (store, params) -> {
                return new Object[]{(int) (Math.random() * 1000)};
            });

            // Compile and instantiate module
            byte[] wasmBytes = loadWasmModule("examples/basic_host_functions.wasm");
            Module module = Module.compile(engine, wasmBytes);

            Store store = Store.create(engine);
            Instance instance = linker.instantiate(store, module);

            // Call WebAssembly function that uses host functions
            WasmFunction main = instance.getExportedFunction("main").orElseThrow();
            main.call(store);

            instance.close();
            store.close();
            module.close();
        }
    }

    /**
     * Advanced module composition with multiple linked modules.
     */
    public static void moduleCompositionExample() {
        Config config = Config.builder().build();

        try (Engine engine = Engine.create(config);
             Linker linker = Linker.create(engine)) {

            // Create utility modules
            Module mathModule = Module.compile(engine, loadWasmModule("examples/math_utils.wasm"));
            Module ioModule = Module.compile(engine, loadWasmModule("examples/io_utils.wasm"));

            Store store = Store.create(engine);

            // Instantiate utility modules
            Instance mathInstance = Instance.create(store, mathModule, new ArrayList<>());
            Instance ioInstance = Instance.create(store, ioModule, new ArrayList<>());

            // Link modules into the linker
            linker.linkInstance("math", mathInstance);
            linker.linkInstance("io", ioInstance);

            // Create aliases for convenience
            linker.alias("app", "sqrt", "math", "sqrt");
            linker.alias("app", "print", "io", "print");

            // Now instantiate the main application module
            Module appModule = Module.compile(engine, loadWasmModule("examples/main_app.wasm"));
            Instance appInstance = linker.instantiate(store, appModule);

            // Run the application
            WasmFunction main = appInstance.getExportedFunction("run").orElseThrow();
            Object result = main.call(store, 42);
            System.out.println("Application result: " + result);

            // Cleanup
            appInstance.close();
            mathInstance.close();
            ioInstance.close();
            appModule.close();
            mathModule.close();
            ioModule.close();
            store.close();
        }
    }

    /**
     * WASI integration example.
     */
    public static void wasiIntegrationExample() {
        Config config = Config.builder()
            .withWasi(true)
            .build();

        try (Engine engine = Engine.create(config);
             Linker linker = Linker.create(engine)) {

            // Configure WASI
            WasiConfig wasiConfig = WasiConfig.builder()
                .withStandardStreams()
                .withEnvironmentVariable("HOME", "/tmp")
                .withArgument("--verbose")
                .withDirectory("/tmp", "/sandbox")
                .build();

            linker.defineWasi(wasiConfig);

            // Additional host functions
            linker.defineHostFunction("env", "custom_log", (store, params) -> {
                String message = params[0].asExternRef().toString();
                System.out.println("[CUSTOM] " + message);
                return new Object[0];
            });

            // Compile and run WASI application
            Module wasiApp = Module.compile(engine, loadWasmModule("examples/wasi_app.wasm"));
            Store store = Store.create(engine);
            Instance instance = linker.instantiate(store, wasiApp);

            WasmFunction start = instance.getExportedFunction("_start").orElseThrow();
            start.call(store);

            instance.close();
            store.close();
            wasiApp.close();
        }
    }

    /**
     * Type introspection and dynamic module analysis.
     */
    public static void typeIntrospectionExample() {
        Config config = Config.builder().build();

        try (Engine engine = Engine.create(config)) {
            byte[] wasmBytes = loadWasmModule("examples/complex_module.wasm");
            Module module = Module.compile(engine, wasmBytes);

            // Analyze module structure
            System.out.println("Module Analysis:");
            System.out.println("================");

            // Analyze imports
            List<ImportDescriptor> imports = module.getImports();
            System.out.println("Imports (" + imports.size() + "):");
            for (ImportDescriptor imp : imports) {
                System.out.printf("  %s::%s - %s%n",
                    imp.getModuleName(), imp.getName(), describeType(imp.getType()));
            }

            // Analyze exports
            List<ExportDescriptor> exports = module.getExports();
            System.out.println("Exports (" + exports.size() + "):");
            for (ExportDescriptor exp : exports) {
                System.out.printf("  %s - %s%n", exp.getName(), describeType(exp.getType()));
            }

            // Analyze function types
            List<FuncType> funcTypes = module.getFunctionTypes();
            System.out.println("Function Types (" + funcTypes.size() + "):");
            for (int i = 0; i < funcTypes.size(); i++) {
                FuncType funcType = funcTypes.get(i);
                System.out.printf("  Type %d: %s -> %s%n", i,
                    funcType.getParameters(), funcType.getResults());
            }

            module.close();
        }
    }

    private static String describeType(WasmType type) {
        switch (type.getKind()) {
            case FUNCTION:
                FuncType funcType = (FuncType) type;
                return String.format("func(%s) -> (%s)",
                    funcType.getParameters(), funcType.getResults());
            case MEMORY:
                MemoryType memType = (MemoryType) type;
                return String.format("memory(min=%d, max=%s, shared=%b)",
                    memType.getMinimumSize(),
                    memType.getMaximumSize().map(Object::toString).orElse("unbounded"),
                    memType.isShared());
            case TABLE:
                TableType tableType = (TableType) type;
                return String.format("table(%s, min=%d, max=%s)",
                    tableType.getElementType(),
                    tableType.getMinimumSize(),
                    tableType.getMaximumSize().map(Object::toString).orElse("unbounded"));
            case GLOBAL:
                GlobalType globalType = (GlobalType) type;
                return String.format("global(%s, mutable=%b)",
                    globalType.getValueType(), globalType.isMutable());
            default:
                return type.toString();
        }
    }

    private static byte[] loadWasmModule(String path) {
        try {
            return Files.readAllBytes(Paths.get(path));
        } catch (IOException e) {
            throw new RuntimeException("Failed to load WASM module: " + path, e);
        }
    }
}

/**
 * Validates 100% API parity with Wasmtime 36.0.2.

public class APIParityValidation {

    private static final WasmtimeAPIReference WASMTIME_REFERENCE = WasmtimeAPIReference.load();

    @TestFactory
    Stream<DynamicTest> validateCompleteAPICoverage() {
        return WASMTIME_REFERENCE.getAllAPIs().stream()
            .map(api -> DynamicTest.dynamicTest(
                "API Parity: " + api.getName(),
                () -> validateAPIImplementation(api)
            ));
    }

    private void validateAPIImplementation(WasmtimeAPI api) {
        // Check that the API is implemented
        assertTrue(isAPIImplemented(api),
            "API not implemented: " + api.getName());

        // Check that the implementation is complete (not a stub)
        assertFalse(isStubImplementation(api),
            "API implementation is a stub: " + api.getName());

        // Validate parameter types and return types
        validateSignatureCompatibility(api);

        // Validate behavior compatibility
        validateBehaviorCompatibility(api);
    }

    @Test
    public void validateWasmtimeFeatureCompleteness() {
        List<String> requiredFeatures = Arrays.asList(
            "Module compilation and validation",
            "Instance creation and management",
            "Function calls (host and WASM)",
            "Memory management and access",
            "Table operations",
            "Global variable access",
            "Import/export handling",
            "Type introspection",
            "Error handling and traps",
            "WASI integration",
            "Linker and dynamic linking",
            "Configuration and optimization",
            "Multi-threading support",
            "Fuel consumption",
            "Epoch-based interruption"
        );

        List<String> missingFeatures = new ArrayList<>();
        for (String feature : requiredFeatures) {
            if (!isFeatureImplemented(feature)) {
                missingFeatures.add(feature);
            }
        }

        assertTrue(missingFeatures.isEmpty(),
            "Missing Wasmtime features: " + String.join(", ", missingFeatures));
    }

    @Test
    public void validateAPIDocumentationCompleteness() {
        List<Class<?>> apiClasses = Arrays.asList(
            Engine.class, Config.class, Module.class, Instance.class, Store.class,
            Linker.class, WasmFunction.class, Memory.class, Table.class, Global.class,
            MemoryType.class, TableType.class, GlobalType.class, FuncType.class,
            ExternFunction.class, ExternMemory.class, ExternTable.class, ExternGlobal.class
        );

        List<String> undocumentedAPIs = new ArrayList<>();
        for (Class<?> apiClass : apiClasses) {
            if (!isClassFullyDocumented(apiClass)) {
                undocumentedAPIs.add(apiClass.getSimpleName());
            }
        }

        assertTrue(undocumentedAPIs.isEmpty(),
            "APIs missing documentation: " + String.join(", ", undocumentedAPIs));
    }

    @Test
    public void validateExampleCompleteness() {
        List<String> requiredExamples = Arrays.asList(
            "Basic module compilation and execution",
            "Host function definition and usage",
            "Memory operations",
            "WASI integration",
            "Module linking and composition",
            "Type introspection",
            "Error handling",
            "Performance optimization"
        );

        List<String> missingExamples = new ArrayList<>();
        for (String example : requiredExamples) {
            if (!isExampleProvided(example)) {
                missingExamples.add(example);
            }
        }

        assertTrue(missingExamples.isEmpty(),
            "Missing examples: " + String.join(", ", missingExamples));
    }

    @Test
    public void validateCrossRuntimeAPIConsistency() {
        List<RuntimeType> availableRuntimes = Arrays.stream(RuntimeType.values())
            .filter(RuntimeType::isAvailable)
            .collect(Collectors.toList());

        if (availableRuntimes.size() < 2) {
            Assumptions.assumeTrue(false, "Need multiple runtimes for consistency testing");
            return;
        }

        // Test all major API operations across runtimes
        List<APIOperation> operations = Arrays.asList(
            new EngineCreationOperation(),
            new ModuleCompilationOperation(),
            new InstanceCreationOperation(),
            new FunctionCallOperation(),
            new MemoryAccessOperation(),
            new TypeIntrospectionOperation(),
            new LinkerOperation(),
            new ErrorHandlingOperation()
        );

        for (APIOperation operation : operations) {
            Map<RuntimeType, Object> results = new HashMap<>();

            for (RuntimeType runtime : availableRuntimes) {
                System.setProperty("wasmtime4j.runtime", runtime.name().toLowerCase());
                Object result = operation.execute();
                results.put(runtime, result);
            }

            validateResultConsistency(operation.getName(), results);
        }
    }

    private void validateResultConsistency(String operationName, Map<RuntimeType, Object> results) {
        if (results.size() < 2) return;

        List<Object> resultsList = new ArrayList<>(results.values());
        Object reference = resultsList.get(0);

        for (int i = 1; i < resultsList.size(); i++) {
            Object current = resultsList.get(i);
            assertTrue(areResultsEquivalent(reference, current),
                String.format("Results differ for operation '%s' between runtimes", operationName));
        }
    }
