# Task: Implement Type Introspection System

## Objective

Implement comprehensive type introspection capabilities by adding MemoryType, TableType, GlobalType, and FuncType interfaces with corresponding native bindings. This enables dynamic WebAssembly module composition and runtime type inspection essential for advanced integration scenarios.

## Implementation Details

### Type System Interfaces

Create complete type introspection interfaces:

```java
// Base interface for all WebAssembly types
public interface WasmType {
    TypeKind getKind();
}

public enum TypeKind {
    FUNCTION,
    MEMORY,
    TABLE,
    GLOBAL
}

// Memory type introspection
public interface MemoryType extends WasmType {
    long getMinimumSize();
    Optional<Long> getMaximumSize();
    boolean isShared();
    boolean is64Bit();

    @Override
    default TypeKind getKind() { return TypeKind.MEMORY; }
}

// Table type introspection
public interface TableType extends WasmType {
    ValueType getElementType();
    long getMinimumSize();
    Optional<Long> getMaximumSize();

    @Override
    default TypeKind getKind() { return TypeKind.TABLE; }
}

// Global type introspection
public interface GlobalType extends WasmType {
    ValueType getValueType();
    boolean isMutable();

    @Override
    default TypeKind getKind() { return TypeKind.GLOBAL; }
}

// Function type introspection
public interface FuncType extends WasmType {
    List<ValueType> getParameters();
    List<ValueType> getResults();

    @Override
    default TypeKind getKind() { return TypeKind.FUNCTION; }
}

// WebAssembly value types
public enum ValueType {
    I32(0),
    I64(1),
    F32(2),
    F64(3),
    V128(4),
    FUNCREF(5),
    EXTERNREF(6);

    private final int value;
    ValueType(int value) { this.value = value; }
    public int getValue() { return value; }

    public static ValueType fromValue(int value) {
        for (ValueType type : values()) {
            if (type.value == value) return type;
        }
        throw new IllegalArgumentException("Unknown value type: " + value);
    }
}
```

### Native Layer Implementation

Add type introspection functions to wasmtime4j-native:

```rust
// Memory type functions
#[no_mangle]
pub extern "C" fn wasmtime4j_memorytype_minimum(memtype_ptr: *mut c_void) -> u64 {
    let memtype = unsafe { &*(memtype_ptr as *const wasmtime::MemoryType) };
    memtype.minimum()
}

#[no_mangle]
pub extern "C" fn wasmtime4j_memorytype_maximum(memtype_ptr: *mut c_void) -> i64 {
    let memtype = unsafe { &*(memtype_ptr as *const wasmtime::MemoryType) };
    memtype.maximum().map(|m| m as i64).unwrap_or(-1)
}

#[no_mangle]
pub extern "C" fn wasmtime4j_memorytype_is_shared(memtype_ptr: *mut c_void) -> bool {
    let memtype = unsafe { &*(memtype_ptr as *const wasmtime::MemoryType) };
    memtype.is_shared()
}

// Function type functions
#[no_mangle]
pub extern "C" fn wasmtime4j_functype_params_count(functype_ptr: *mut c_void) -> u32 {
    let functype = unsafe { &*(functype_ptr as *const wasmtime::FuncType) };
    functype.params().len() as u32
}

#[no_mangle]
pub extern "C" fn wasmtime4j_functype_param_at(functype_ptr: *mut c_void, index: u32) -> i32 {
    let functype = unsafe { &*(functype_ptr as *const wasmtime::FuncType) };
    if let Some(param) = functype.params().nth(index as usize) {
        wasmtime_valtype_to_int(param)
    } else {
        -1
    }
}

// Module introspection functions
#[no_mangle]
pub extern "C" fn wasmtime4j_module_imports_count(module_ptr: *mut c_void) -> u32 {
    let module = unsafe { &*(module_ptr as *const wasmtime::Module) };
    module.imports().count() as u32
}

#[no_mangle]
pub extern "C" fn wasmtime4j_module_import_at(
    module_ptr: *mut c_void,
    index: u32,
    module_name_out: *mut *mut c_char,
    name_out: *mut *mut c_char,
    type_kind_out: *mut i32
) -> bool {
    let module = unsafe { &*(module_ptr as *const wasmtime::Module) };
    if let Some(import) = module.imports().nth(index as usize) {
        unsafe {
            *module_name_out = CString::new(import.module()).unwrap().into_raw();
            *name_out = CString::new(import.name()).unwrap().into_raw();
            *type_kind_out = wasmtime_extern_type_to_int(import.ty());
        }
        true
    } else {
        false
    }
}

// Helper functions for type conversion
fn wasmtime_valtype_to_int(valtype: &wasmtime::ValType) -> i32 {
    match valtype {
        wasmtime::ValType::I32 => 0,
        wasmtime::ValType::I64 => 1,
        wasmtime::ValType::F32 => 2,
        wasmtime::ValType::F64 => 3,
        wasmtime::ValType::V128 => 4,
        wasmtime::ValType::FuncRef => 5,
        wasmtime::ValType::ExternRef => 6,
    }
}
```

### JNI Implementation

Implement type introspection in JNI:

```java
public final class JniMemoryType implements MemoryType {
    private final long nativeHandle;

    public JniMemoryType(long nativeHandle) {
        this.nativeHandle = nativeHandle;
    }

    @Override
    public long getMinimumSize() {
        return nativeGetMinimum(nativeHandle);
    }

    @Override
    public Optional<Long> getMaximumSize() {
        long max = nativeGetMaximum(nativeHandle);
        return max >= 0 ? Optional.of(max) : Optional.empty();
    }

    @Override
    public boolean isShared() {
        return nativeIsShared(nativeHandle);
    }

    @Override
    public boolean is64Bit() {
        return nativeIs64Bit(nativeHandle);
    }

    // Native method declarations
    private static native long nativeGetMinimum(long handle);
    private static native long nativeGetMaximum(long handle);
    private static native boolean nativeIsShared(long handle);
    private static native boolean nativeIs64Bit(long handle);
}

public final class JniModule implements Module {
    // Existing implementation...

    @Override
    public List<ImportDescriptor> getImports() {
        int count = nativeGetImportsCount(nativeHandle);
        List<ImportDescriptor> imports = new ArrayList<>(count);

        for (int i = 0; i < count; i++) {
            String[] names = new String[2];
            int[] typeKind = new int[1];
            long typeHandle = nativeGetImportAt(nativeHandle, i, names, typeKind);

            ImportDescriptor descriptor = new ImportDescriptor() {
                @Override
                public String getModuleName() { return names[0]; }

                @Override
                public String getName() { return names[1]; }

                @Override
                public WasmType getType() {
                    return createTypeFromHandle(TypeKind.fromValue(typeKind[0]), typeHandle);
                }
            };
            imports.add(descriptor);
        }

        return Collections.unmodifiableList(imports);
    }

    // Native method declarations
    private static native int nativeGetImportsCount(long handle);
    private static native long nativeGetImportAt(long handle, int index, String[] names, int[] typeKind);
}
```
