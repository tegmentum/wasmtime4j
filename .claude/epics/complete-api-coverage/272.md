// GC-managed WebAssembly values
public abstract class GCManagedValue extends WasmValue {
    public abstract boolean isNull();
    public abstract GCObjectType getGCType();
    public abstract long getObjectId();

    // Type casting for GC types
    public WasmStruct asStruct() {
        if (getGCType() != GCObjectType.STRUCT) {
            throw new ClassCastException("Not a struct type");
        }
        return (WasmStruct) this;
    }

    public WasmArray asArray() {
        if (getGCType() != GCObjectType.ARRAY) {
            throw new ClassCastException("Not an array type");
        }
        return (WasmArray) this;
    }
}

// WebAssembly struct implementation
public class WasmStruct extends GCManagedValue {
    private final StructType structType;
    private final Map<String, WasmValue> fields;

    public WasmStruct(StructType type, Map<String, WasmValue> initialFields) {
        this.structType = type;
        this.fields = new HashMap<>(initialFields);
    }

    public WasmValue getField(String name) {
        return fields.get(name);
    }

    public void setField(String name, WasmValue value) {
        if (structType.isFieldMutable(name)) {
            fields.put(name, value);
        } else {
            throw new IllegalArgumentException("Field " + name + " is immutable");
        }
    }

    public Set<String> getFieldNames() {
        return structType.getFieldNames();
    }

    @Override
    public GCObjectType getGCType() {
        return GCObjectType.STRUCT;
    }
}

// WebAssembly array implementation
public class WasmArray extends GCManagedValue {
    private final ArrayType arrayType;
    private final List<WasmValue> elements;

    public WasmArray(ArrayType type, int initialSize) {
        this.arrayType = type;
        this.elements = new ArrayList<>(Collections.nCopies(initialSize, type.getElementType().getDefaultValue()));
    }

    public WasmValue get(int index) {
        return elements.get(index);
    }

    public void set(int index, WasmValue value) {
        if (arrayType.isElementMutable()) {
            elements.set(index, value);
        } else {
            throw new IllegalArgumentException("Array elements are immutable");
        }
    }

    public int size() {
        return elements.size();
    }

    public void resize(int newSize) {
        if (newSize > elements.size()) {
            WasmValue defaultValue = arrayType.getElementType().getDefaultValue();
            while (elements.size() < newSize) {
                elements.add(defaultValue);
            }
        } else {
            elements.subList(newSize, elements.size()).clear();
        }
    }

    @Override
    public GCObjectType getGCType() {
        return GCObjectType.ARRAY;
    }
}

        super("WebAssembly exception: " + type.getExceptionName());
        this.exceptionType = type;
        this.payload = List.copyOf(payload);
        this.stackTrace = WasmStackTrace.capture();
    }

    public WasmExceptionType getExceptionType() {
        return exceptionType;
    }

    public List<WasmValue> getPayload() {
        return payload;
    }

    public WasmStackTrace getWasmStackTrace() {
        return stackTrace;
    }

    // Payload access helpers
    public <T> T getPayloadValue(int index, Class<T> type) {
        if (index >= payload.size()) {
            throw new IndexOutOfBoundsException("Payload index out of bounds");
        }
        WasmValue value = payload.get(index);
        return value.getValue(type);
    }
}

// Function reference types and operations
public class WasmFuncRef extends WasmValue {
    private final Optional<WasmFunction> function;
    private final FunctionType signature;

    public static WasmFuncRef nullRef(FunctionType signature) {
        return new WasmFuncRef(Optional.empty(), signature);
    }

    public static WasmFuncRef of(WasmFunction function) {
        return new WasmFuncRef(Optional.of(function), function.getType());
    }

    private WasmFuncRef(Optional<WasmFunction> function, FunctionType signature) {
        this.function = function;
        this.signature = signature;
    }

    public boolean isNull() {
        return function.isEmpty();
    }

    public Optional<WasmFunction> getFunction() {
        return function;
    }

    public FunctionType getSignature() {
        return signature;
    }

    // Function reference calling
    public WasmValue call(Store store, WasmValue... parameters) {
        if (isNull()) {
            throw new WasmException("Cannot call null function reference");
        }
        return function.get().call(store, parameters);
    }

    public CompletableFuture<WasmValue> callAsync(Store store, WasmValue... parameters) {
        if (isNull()) {
            return CompletableFuture.failedFuture(new WasmException("Cannot call null function reference"));
        }
        return function.get().callAsync(store, parameters);
    }
}
