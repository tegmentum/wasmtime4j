```java
/**
 * Example: Weather Service Component
 *
 * WIT Definition:
 * interface weather-service {
 *   record location {
 *     latitude: f64,
 *     longitude: f64
 *   }
 *
 *   variant weather-condition {
 *     sunny,
 *     cloudy,
 *     rainy(f64), // mm of rain
 *     snowy(f64)  // cm of snow
 *   }
 *
 *   get-weather: func(loc: location) -> weather-condition;
 * }
 */
public class WeatherServiceExample {

    public void exampleUsage() {
        // Parse WIT interface
        ComponentInterface weatherInterface = WitParser.parseInterface(
            Paths.get("interfaces/weather-service.wit"));

        // Create component from interface
        Component weatherComponent = Component.compile(engine, weatherWasmBytes);

        try (ComponentInstance instance = weatherComponent.instantiate(store, linker)) {
            // Create location record
            Map<String, ComponentValue> locationFields = Map.of(
                "latitude", ComponentValue.of(40.7128),
                "longitude", ComponentValue.of(-74.0060)
            );
            ComponentRecord location = new ComponentRecord(locationFields);

            // Call weather function
            ComponentFunction getWeather = instance
                .getFunction("weather-service", "get-weather")
                .orElseThrow();

            ComponentValue result = getWeather.call(store, location);
            if (result instanceof ComponentVariant variant) {
                switch (variant.getCaseName()) {
                    case "sunny" -> System.out.println("It's sunny!");
                    case "rainy" -> System.out.println("Rain: " + variant.getValue().getValue() + "mm");
                    // ... handle other cases
                }
            }
        }
    }
}
```

```java
        // Create parameter stream

        // Process with backpressure
        Publisher<Object> resultStream = processor.callReactive(store, inputStream,
                                                              FlowControlStrategy.BUFFER);

        // Subscribe to results
        resultStream.subscribe(new Subscriber<Object>() {
            @Override
            public void onNext(Object result) {
                processResult(result);

                handleError(error);
```

### Native Layer Extensions for Async Support

Extend native layer with async capabilities:

```rust
// Async function call support
#[no_mangle]
pub extern "C" fn wasmtime4j_func_call_async(
    func_ptr: *mut c_void,
    store_ptr: *mut c_void,
    params_ptr: *const *const c_void,
    params_count: u32,
    callback_id: usize
) -> bool {
    let func = unsafe { &*(func_ptr as *const wasmtime::Func) };
    let mut store = unsafe { &mut *(store_ptr as *mut wasmtime::Store<()>) };

    // Convert parameters
    let params = convert_params(params_ptr, params_count);

    // Execute async call
    tokio::spawn(async move {
        match func.call_async(&mut store, &params, &mut []).await {
            Ok(results) => {
                notify_callback_success(callback_id, results);
            }
            Err(error) => {
                notify_callback_error(callback_id, error);
            }
        }
    });

    true
}

// Component model support
#[no_mangle]
pub extern "C" fn wasmtime4j_component_compile(
    engine_ptr: *mut c_void,
    component_bytes: *const u8,
    bytes_len: u32
) -> *mut c_void {
    let engine = unsafe { &*(engine_ptr as *const wasmtime::Engine) };
    let bytes = unsafe { std::slice::from_raw_parts(component_bytes, bytes_len as usize) };

    match wasmtime::component::Component::new(engine, bytes) {
        Ok(component) => Box::into_raw(Box::new(component)) as *mut c_void,
        Err(_) => std::ptr::null_mut(),
    }
}

#[no_mangle]
pub extern "C" fn wasmtime4j_component_instantiate_async(
    component_ptr: *mut c_void,
    store_ptr: *mut c_void,
    linker_ptr: *mut c_void,
    callback_id: usize
) -> bool {
    let component = unsafe { &*(component_ptr as *const wasmtime::component::Component) };
    let store = unsafe { &mut *(store_ptr as *mut wasmtime::Store<()>) };
    let linker = unsafe { &*(linker_ptr as *const wasmtime::component::Linker<()>) };

    tokio::spawn(async move {
        match linker.instantiate_async(store, component).await {
            Ok(instance) => {
                notify_component_callback_success(callback_id, instance);
            }
            Err(error) => {
                notify_component_callback_error(callback_id, error);
            }
        }
    });

    true
}
```
