// Base class for all component values
public abstract class ComponentValue {

    public static ComponentValue of(Object value) {
        // Factory method for creating ComponentValue from Java objects
    }

    public abstract Object getValue();
    public abstract ComponentValueType getType();

    // Type checking methods
    public boolean isRecord() { return this instanceof ComponentRecord; }
    public boolean isVariant() { return this instanceof ComponentVariant; }
    public boolean isList() { return this instanceof ComponentList; }
    public boolean isPrimitive() { return this instanceof ComponentPrimitive; }

    // Casting methods with validation
    public ComponentRecord asRecord() {
        if (!isRecord()) throw new IllegalStateException("Not a record type");
        return (ComponentRecord) this;
    }

    public ComponentVariant asVariant() {
        if (!isVariant()) throw new IllegalStateException("Not a variant type");
        return (ComponentVariant) this;
    }
}

// Record type (struct-like)
public class ComponentRecord extends ComponentValue {
    private final Map<String, ComponentValue> fields;
    private final ComponentRecordType recordType;

    public ComponentRecord(ComponentRecordType type, Map<String, ComponentValue> fields) {
        this.recordType = type;
        this.fields = Map.copyOf(fields);
    }

    public ComponentValue getField(String name) {
        return fields.get(name);
    }

    public Set<String> getFieldNames() {
        return fields.keySet();
    }

    public ComponentRecordType getRecordType() {
        return recordType;
    }

    // Builder pattern for construction
    public static Builder builder(ComponentRecordType type) {
        return new Builder(type);
    }

    public static class Builder {
        private final ComponentRecordType type;
        private final Map<String, ComponentValue> fields = new HashMap<>();

        public Builder field(String name, ComponentValue value) {
            fields.put(name, value);
            return this;
        }

        public ComponentRecord build() {
            // Validate all required fields are present
            return new ComponentRecord(type, fields);
        }
    }
}

// Variant type (union-like)
public class ComponentVariant extends ComponentValue {
    private final String caseName;
    private final ComponentValue value;
    private final ComponentVariantType variantType;

    public ComponentVariant(ComponentVariantType type, String caseName, ComponentValue value) {
        this.variantType = type;
        this.caseName = caseName;
        this.value = value;
    }

    public String getCaseName() { return caseName; }
    public ComponentValue getValue() { return value; }
    public ComponentVariantType getVariantType() { return variantType; }

    // Pattern matching support
    public <T> T match(Map<String, Function<ComponentValue, T>> cases) {
        Function<ComponentValue, T> handler = cases.get(caseName);
        if (handler == null) {
            throw new IllegalArgumentException("No handler for case: " + caseName);
        }
        return handler.apply(value);
    }
}

// List type
public class ComponentList extends ComponentValue {
    private final List<ComponentValue> elements;
    private final ComponentListType listType;

    public ComponentList(ComponentListType type, List<ComponentValue> elements) {
        this.listType = type;
        this.elements = List.copyOf(elements);
    }

    public List<ComponentValue> getElements() { return elements; }
    public int size() { return elements.size(); }
    public ComponentValue get(int index) { return elements.get(index); }
    public ComponentListType getListType() { return listType; }

    // Stream operations
    public Stream<ComponentValue> stream() { return elements.stream(); }
}

// Primitive types
public class ComponentPrimitive extends ComponentValue {
    private final Object value;

    public int asS32() { return (Integer) value; }
    public long asS64() { return (Long) value; }

/**
 * Example: Weather Service Component Integration
 *
 * WIT Definition (weather.wit):
 * ```
 * package weather:service@1.0.0;
 *
 * interface weather {
 *   record location {
 *     latitude: float64,
 *     longitude: float64,
 *     name: string
 *   }
 *
 *   variant condition {
 *     sunny,
 *     cloudy(u8),           // cloud cover percentage
 *     rainy(float32),       // mm of rain
 *     snowy(float32)        // cm of snow
 *   }
 *
 *   resource forecast {
 *     constructor(loc: location);
 *     get-current: func() -> condition;
 *     get-hourly: func(hours: u32) -> list<condition>;
 *   }
 *
 *   get-weather: func(loc: location) -> condition;
 *   create-forecast: func(loc: location) -> forecast;
 * }
 * ```
 */
public class WeatherServiceExample {

    public void demonstrateComponentModel() throws Exception {
        // Parse WIT interface
        ComponentInterface weatherInterface = WitParser.parseInterface(
            Paths.get("interfaces/weather.wit"));

        // Compile component
        Component weatherComponent = Component.compile(engine,
            Files.readAllBytes(Paths.get("weather-service.wasm")));

        try (ComponentLinker linker = ComponentLinker.create(engine);
             Store store = Store.create(engine)) {

            // Create component instance
            ComponentInstance instance = linker.instantiate(store, weatherComponent);

            // Create location record
            ComponentRecordType locationType = weatherInterface.getType("location").asRecord();
            ComponentRecord location = ComponentRecord.builder(locationType)
                .field("latitude", ComponentPrimitive.float64(40.7128))
                .field("longitude", ComponentPrimitive.float64(-74.0060))
                .field("name", ComponentPrimitive.string("New York"))
                .build();

            // Call weather function
            ComponentFunction getWeather = instance
                .getFunction("weather", "get-weather")
                .orElseThrow();

            ComponentValue result = getWeather.call(store, location);

            // Pattern match on variant result
            if (result instanceof ComponentVariant condition) {
                String forecast = condition.match(Map.of(
                    "sunny", v -> "It's sunny!",
                    "cloudy", v -> "Cloudy with " + v.asPrimitive().asU8() + "% coverage",
                    "rainy", v -> "Rain: " + v.asPrimitive().asFloat32() + "mm",
                    "snowy", v -> "Snow: " + v.asPrimitive().asFloat32() + "cm"
                ));
                System.out.println(forecast);
            }

            // Work with resources
            ComponentFunction createForecast = instance
                .getFunction("weather", "create-forecast")
                .orElseThrow();

            ComponentValue forecastResource = createForecast.call(store, location);

            // Use resource methods
            if (forecastResource instanceof ComponentResource<?> forecast) {
                ComponentFunction getCurrent = instance
                    .getFunction("weather", "get-current")
                    .orElseThrow();

                ComponentValue currentWeather = getCurrent.call(store);
                System.out.println("Current weather: " + currentWeather);
            }
        }
    }
}

    engine_ptr: *mut c_void,
    component_bytes: *const u8,
    bytes_len: u32
) -> *mut c_void {
    let engine = unsafe { &*(engine_ptr as *const wasmtime::Engine) };
    let bytes = unsafe { std::slice::from_raw_parts(component_bytes, bytes_len as usize) };

    match Component::new(engine, bytes) {
        Ok(component) => Box::into_raw(Box::new(component)) as *mut c_void,
        Err(_) => std::ptr::null_mut(),
    }

        Ok(instance) => Box::into_raw(Box::new(instance)) as *mut c_void,
        Err(_) => std::ptr::null_mut(),
