    config.wasm_garbage_collection(enabled).is_ok()
}

#[no_mangle]
pub extern "C" fn wasmtime4j_config_set_wasm_exception_handling(
    config_ptr: *mut c_void,
    enabled: bool
) -> bool {
    let config = unsafe { &mut *(config_ptr as *mut wasmtime::Config) };
    config.wasm_exception_handling(enabled).is_ok()
}

#[no_mangle]
pub extern "C" fn wasmtime4j_config_set_compilation_strategy(
    config_ptr: *mut c_void,
    strategy: i32
) -> bool {
    let config = unsafe { &mut *(config_ptr as *mut wasmtime::Config) };
    let strategy = match strategy {
        0 => wasmtime::Strategy::Auto,
        1 => wasmtime::Strategy::Cranelift,
        2 => wasmtime::Strategy::Winch,
        _ => return false,
    };
    config.strategy(strategy).is_ok()
}

#[no_mangle]
pub extern "C" fn wasmtime4j_config_set_profiling_strategy(
    config_ptr: *mut c_void,
    strategy: i32
) -> bool {
    let config = unsafe { &mut *(config_ptr as *mut wasmtime::Config) };
    let profiler = match strategy {
        0 => wasmtime::ProfilingStrategy::None,
        1 => wasmtime::ProfilingStrategy::JitDump,
        2 => wasmtime::ProfilingStrategy::VTune,
        3 => wasmtime::ProfilingStrategy::PerfMap,
        _ => return false,
    };
    config.profiler(profiler).is_ok()
}

#[no_mangle]
pub extern "C" fn wasmtime4j_config_set_memory_guard_size(
    config_ptr: *mut c_void,
    guard_size: u64
) -> bool {
    let config = unsafe { &mut *(config_ptr as *mut wasmtime::Config) };
    config.static_memory_guard_size(guard_size).is_ok()
}

#[no_mangle]
pub extern "C" fn wasmtime4j_config_set_parallel_compilation(
    config_ptr: *mut c_void,
    enabled: bool
) -> bool {
    let config = unsafe { &mut *(config_ptr as *mut wasmtime::Config) };
    config.parallel_compilation(enabled).is_ok()
}
