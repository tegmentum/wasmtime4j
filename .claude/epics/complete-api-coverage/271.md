### Native Layer Performance Monitoring

```rust
// Performance monitoring in native layer
use std::time::{Instant, Duration};
use std::collections::HashMap;

// Global performance tracking
static mut PERFORMANCE_TRACKER: Option<PerformanceTracker> = None;

struct PerformanceTracker {
    function_calls: HashMap<String, CallStatistics>,
    compilation_stats: CompilationStats,
    memory_stats: MemoryStats,
    start_time: Instant,
}

struct CallStatistics {
    count: u64,
    total_time: Duration,
    min_time: Duration,
    max_time: Duration,
}

struct CompilationStats {
    modules_compiled: u64,
    total_compilation_time: Duration,
    cache_hits: u64,
    cache_misses: u64,
}

struct MemoryStats {
    total_allocated: u64,
    total_deallocated: u64,
    peak_usage: u64,
    current_usage: u64,
}

#[no_mangle]
pub extern "C" fn wasmtime4j_perf_start_tracking() -> bool {
    unsafe {
        if PERFORMANCE_TRACKER.is_none() {
            PERFORMANCE_TRACKER = Some(PerformanceTracker {
                function_calls: HashMap::new(),
                compilation_stats: CompilationStats {
                    modules_compiled: 0,
                    total_compilation_time: Duration::ZERO,
                    cache_hits: 0,
                    cache_misses: 0,
                },
                memory_stats: MemoryStats {
                    total_allocated: 0,
                    total_deallocated: 0,
                    peak_usage: 0,
                    current_usage: 0,
                },
                start_time: Instant::now(),
            });
            true
        } else {
            false
        }
    }
}

#[no_mangle]
pub extern "C" fn wasmtime4j_perf_record_function_call(
    function_name: *const c_char,
    execution_time_ns: u64
) {
    unsafe {
        if let Some(ref mut tracker) = PERFORMANCE_TRACKER {
            let name = CStr::from_ptr(function_name).to_str().unwrap_or("unknown").to_string();
            let duration = Duration::from_nanos(execution_time_ns);

            let stats = tracker.function_calls.entry(name).or_insert(CallStatistics {
                count: 0,
                total_time: Duration::ZERO,
                min_time: Duration::MAX,
                max_time: Duration::ZERO,
            });

            stats.count += 1;
            stats.total_time += duration;
            stats.min_time = stats.min_time.min(duration);
            stats.max_time = stats.max_time.max(duration);
        }
    }
}

#[no_mangle]
pub extern "C" fn wasmtime4j_perf_record_compilation(
    compilation_time_ns: u64,
    cache_hit: bool
) {
    unsafe {
        if let Some(ref mut tracker) = PERFORMANCE_TRACKER {
            tracker.compilation_stats.modules_compiled += 1;
            tracker.compilation_stats.total_compilation_time += Duration::from_nanos(compilation_time_ns);

            if cache_hit {
                tracker.compilation_stats.cache_hits += 1;
            } else {
                tracker.compilation_stats.cache_misses += 1;
            }
        }
    }
}

#[no_mangle]
pub extern "C" fn wasmtime4j_perf_record_memory_allocation(size: u64) {
    unsafe {
        if let Some(ref mut tracker) = PERFORMANCE_TRACKER {
            tracker.memory_stats.total_allocated += size;
            tracker.memory_stats.current_usage += size;
            tracker.memory_stats.peak_usage = tracker.memory_stats.peak_usage.max(tracker.memory_stats.current_usage);
        }
    }
}

#[no_mangle]
pub extern "C" fn wasmtime4j_perf_record_memory_deallocation(size: u64) {
    unsafe {
        if let Some(ref mut tracker) = PERFORMANCE_TRACKER {
            tracker.memory_stats.total_deallocated += size;
            tracker.memory_stats.current_usage = tracker.memory_stats.current_usage.saturating_sub(size);
        }
    }
}

#[no_mangle]
pub extern "C" fn wasmtime4j_perf_get_statistics(stats_ptr: *mut PerformanceStatsNative) -> bool {
    unsafe {
        if let Some(ref tracker) = PERFORMANCE_TRACKER {
            let stats = &mut *stats_ptr;

            stats.uptime_ms = tracker.start_time.elapsed().as_millis() as u64;
            stats.total_function_calls = tracker.function_calls.values().map(|s| s.count).sum();
            stats.modules_compiled = tracker.compilation_stats.modules_compiled;
            stats.total_memory_allocated = tracker.memory_stats.total_allocated;
            stats.peak_memory_usage = tracker.memory_stats.peak_usage;
            stats.current_memory_usage = tracker.memory_stats.current_usage;

            true
        } else {
            false
        }
    }
}

#[repr(C)]
pub struct PerformanceStatsNative {
    pub uptime_ms: u64,
    pub total_function_calls: u64,
    pub modules_compiled: u64,
    pub total_memory_allocated: u64,
    pub peak_memory_usage: u64,
    pub current_memory_usage: u64,
}
```
