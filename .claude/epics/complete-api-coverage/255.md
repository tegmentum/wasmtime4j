```rust
    unsafe {
```

```rust
// src/error.rs - Centralized error handling
use wasmtime::*;
use std::ffi::CString;

pub struct ErrorInfo {
    pub error_type: i32,
    pub message: CString,
    pub details: Option<CString>,
}

// Error type constants for Java mapping
pub const ERROR_COMPILATION: i32 = 1;
pub const ERROR_RUNTIME: i32 = 2;
pub const ERROR_VALIDATION: i32 = 3;
pub const ERROR_INSTANTIATION: i32 = 4;
pub const ERROR_LINKING: i32 = 5;

// Convert Wasmtime errors to structured error info
pub fn map_wasmtime_error(error: &anyhow::Error) -> ErrorInfo {
    if let Some(trap) = error.downcast_ref::<Trap>() {
        ErrorInfo {
            error_type: ERROR_RUNTIME,
            message: CString::new(format!("Runtime trap: {}", trap)).unwrap(),
            details: Some(CString::new(format!("{:?}", trap)).unwrap()),
        }
    } else if error.to_string().contains("validation") {
        ErrorInfo {
            error_type: ERROR_VALIDATION,
            message: CString::new(format!("Validation error: {}", error)).unwrap(),
            details: None,
        }
    } else if error.to_string().contains("compilation") {
        ErrorInfo {
            error_type: ERROR_COMPILATION,
            message: CString::new(format!("Compilation error: {}", error)).unwrap(),
            details: None,
        }
    } else {
        ErrorInfo {
            error_type: ERROR_RUNTIME,
            message: CString::new(format!("Unknown error: {}", error)).unwrap(),
            details: None,
        }
    }
}

// Global error state for last error
static mut LAST_ERROR: Option<ErrorInfo> = None;

#[no_mangle]
pub extern "C" fn wasmtime4j_get_last_error(
    error_type_out: *mut i32,
    message_out: *mut *const c_char,
    details_out: *mut *const c_char
) -> bool {
    unsafe {
        if let Some(ref error) = LAST_ERROR {
            *error_type_out = error.error_type;
            *message_out = error.message.as_ptr();
            *details_out = error.details.as_ref()
                .map(|d| d.as_ptr())
                .unwrap_or(std::ptr::null());
            true
        } else {
            false
        }
    }
}

#[no_mangle]
pub extern "C" fn wasmtime4j_clear_last_error() {
    unsafe {
        LAST_ERROR = None;
    }
}
```

```rust
// Configuration introspection support
#[no_mangle]
pub extern "C" fn wasmtime4j_config_register(config_ptr: *mut c_void) -> usize {
    let config = unsafe { Box::from_raw(config_ptr as *mut Config) };
    unsafe {
        if let Some(ref mut registry) = WASMTIME_CONFIG_REGISTRY {
            let id = registry.next_id;
            registry.configs.insert(id, *config);
            registry.next_id += 1;
            id
        } else {
            0
        }
    }
}

#[no_mangle]
pub extern "C" fn wasmtime4j_config_get_optimization_level(config_id: usize) -> i32 {
    unsafe {
        if let Some(ref registry) = WASMTIME_CONFIG_REGISTRY {
            if let Some(config) = registry.configs.get(&config_id) {
                // Note: Wasmtime doesn't expose these values directly
                // We need to track them during configuration
                return 1; // Default to Speed
            }
        }
    }
    -1
}
```

```rust
// src/performance.rs - Performance optimizations
use std::sync::Arc;
use std::collections::HashMap;

// Function call optimization with caching
static mut FUNCTION_CACHE: Option<Arc<std::sync::Mutex<HashMap<usize, Func>>>> = None;

#[no_mangle]
pub extern "C" fn wasmtime4j_func_call_optimized(
    func_ptr: *mut c_void,
    store_ptr: *mut c_void,
    params_ptr: *const *const c_void,
    params_count: u32,
    results_ptr: *mut *mut c_void,
    results_capacity: u32
) -> i32 {
    let func = unsafe { &*(func_ptr as *const Func) };
    let mut store = unsafe { &mut *(store_ptr as *mut Store<()>) };

    // Convert parameters efficiently
    let mut params = Vec::with_capacity(params_count as usize);
    for i in 0..params_count {
        let param_ptr = unsafe { *params_ptr.add(i as usize) };
        // Convert based on function signature
        let val = unsafe { convert_param_from_ptr(param_ptr, func.ty(&store).params().nth(i as usize)) };
        params.push(val);
    }

    // Prepare results buffer
    let results_count = func.ty(&store).results().len();
    let mut results = vec![Val::I32(0); results_count];

    // Call function with optimized path
    match func.call(&mut store, &params, &mut results) {
        Ok(_) => {
            // Convert results efficiently
            for (i, result) in results.iter().enumerate() {
                if i < results_capacity as usize {
                    unsafe {
                        *results_ptr.add(i) = convert_val_to_ptr(result);
                    }
                }
            }
            results_count as i32
        }
        Err(e) => {
            unsafe {
                LAST_ERROR = Some(map_wasmtime_error(&e.into()));
            }
            -1
        }
    }
}

// Memory operation optimizations
#[no_mangle]
pub extern "C" fn wasmtime4j_memory_bulk_copy(
    memory_ptr: *mut c_void,
    store_ptr: *mut c_void,
    dst_offset: u32,
    src_ptr: *const u8,
    size: u32
) -> bool {
    let memory = unsafe { &*(memory_ptr as *const Memory) };
    let mut store = unsafe { &mut *(store_ptr as *mut Store<()>) };
    let src_slice = unsafe { std::slice::from_raw_parts(src_ptr, size as usize) };

    match memory.write(&mut store, dst_offset as usize, src_slice) {
        Ok(_) => true,
        Err(e) => {
            unsafe {
                LAST_ERROR = Some(map_wasmtime_error(&e.into()));
            }
            false
        }
    }
}
```
