// src/main/java/ai/tegmentum/wasmtime4j/performance/PerformanceOptimizer.java
public class PerformanceOptimizer {

    // Function call optimization
    public static class FunctionCallOptimizer {
        private static final Map<String, MethodHandle> cachedMethodHandles = new ConcurrentHashMap<>();
        private static final Map<FuncType, ParameterConverter> converterCache = new ConcurrentHashMap<>();

        public static Object optimizedCall(WasmFunction function, Store store, Object... params) {
            FuncType funcType = function.getType();
            ParameterConverter converter = converterCache.computeIfAbsent(funcType,
                type -> new ParameterConverter(type));

            // Use pre-compiled parameter conversion
            Val[] wasmParams = converter.convertParameters(params);

            // Optimized native call path
            return function.callOptimized(store, wasmParams);
        }
    }

    // Memory access optimization
    public static class MemoryAccessOptimizer {
        private static final ThreadLocal<ByteBuffer> bufferCache = ThreadLocal.withInitial(
            () -> ByteBuffer.allocateDirect(8192)
        );

        public static void bulkRead(Memory memory, Store store, long offset, byte[] data) {
            if (data.length <= 8192) {
                // Use cached direct buffer for small reads
                ByteBuffer buffer = bufferCache.get();
                buffer.clear();
                buffer.limit(data.length);

                memory.readDirect(store, offset, buffer);
                buffer.get(data);
            } else {
                // Use streaming approach for large reads
                memory.readStreaming(store, offset, data);
            }
        }

        public static void bulkWrite(Memory memory, Store store, long offset, byte[] data) {
            if (data.length <= 8192) {
                ByteBuffer buffer = bufferCache.get();
                buffer.clear();
                buffer.put(data);
                buffer.flip();

                memory.writeDirect(store, offset, buffer);
            } else {
                memory.writeStreaming(store, offset, data);
            }
        }
    }

    // Type introspection optimization
    public static class TypeIntrospectionOptimizer {
        private static final Map<Module, ModuleTypeCache> typeCache = new ConcurrentHashMap<>();

        public static class ModuleTypeCache {
            private final List<ImportDescriptor> imports;
            private final List<ExportDescriptor> exports;
            private final List<FuncType> functionTypes;

            public ModuleTypeCache(Module module) {
                // Pre-compute and cache all type information
                this.imports = Collections.unmodifiableList(computeImports(module));
                this.exports = Collections.unmodifiableList(computeExports(module));
                this.functionTypes = Collections.unmodifiableList(computeFunctionTypes(module));
            }

            // Fast lookup methods
            public List<ImportDescriptor> getImports() { return imports; }
            public List<ExportDescriptor> getExports() { return exports; }
            public List<FuncType> getFunctionTypes() { return functionTypes; }
        }

        public static ModuleTypeCache getTypeCache(Module module) {
            return typeCache.computeIfAbsent(module, ModuleTypeCache::new);
        }
    }
}

public class JniPerformanceOptimizations {

    // Optimized parameter passing
    public static class ParameterPassingOptimizer {
        // Pre-allocated parameter arrays for common function signatures
        private static final ThreadLocal<long[]> longParamCache = ThreadLocal.withInitial(() -> new long[8]);
        private static final ThreadLocal<double[]> doubleParamCache = ThreadLocal.withInitial(() -> new double[8]);

        public static Object callFunctionOptimized(long functionHandle, long storeHandle, Object... params) {
            if (params.length <= 8 && areAllIntegers(params)) {
                // Use optimized integer call path
                long[] longParams = longParamCache.get();
                for (int i = 0; i < params.length; i++) {
                    longParams[i] = ((Number) params[i]).longValue();
                }
                return nativeCallWithLongs(functionHandle, storeHandle, longParams, params.length);
            } else if (params.length <= 8 && areAllFloats(params)) {
                // Use optimized float call path
                double[] doubleParams = doubleParamCache.get();
                for (int i = 0; i < params.length; i++) {
                    doubleParams[i] = ((Number) params[i]).doubleValue();
                }
                return nativeCallWithDoubles(functionHandle, storeHandle, doubleParams, params.length);
            } else {
                // Fall back to general call path
                return nativeCallGeneral(functionHandle, storeHandle, params);
            }
        }

        private static native Object nativeCallWithLongs(long funcHandle, long storeHandle, long[] params, int count);
        private static native Object nativeCallWithDoubles(long funcHandle, long storeHandle, double[] params, int count);
        private static native Object nativeCallGeneral(long funcHandle, long storeHandle, Object[] params);
    }

    // Memory access optimization
    public static class MemoryAccessOptimizer {
        // Direct buffer operations for better performance
        public static void readDirect(long memoryHandle, long storeHandle, long offset, ByteBuffer buffer) {
            nativeReadDirect(memoryHandle, storeHandle, offset, buffer, buffer.remaining());
        }

        public static void writeDirect(long memoryHandle, long storeHandle, long offset, ByteBuffer buffer) {
            nativeWriteDirect(memoryHandle, storeHandle, offset, buffer, buffer.remaining());
        }

        private static native void nativeReadDirect(long memHandle, long storeHandle, long offset, ByteBuffer buffer, int size);
        private static native void nativeWriteDirect(long memHandle, long storeHandle, long offset, ByteBuffer buffer, int size);
    }

    // String handling optimization
    public static class StringOptimizer {
        // Cache for frequently used strings
        private static final Map<String, byte[]> utf8Cache = new ConcurrentHashMap<>();

        public static byte[] getUTF8Bytes(String str) {
            return utf8Cache.computeIfAbsent(str, s -> s.getBytes(StandardCharsets.UTF_8));
        }
    }
}

// src/main/java/ai/tegmentum/wasmtime4j/panama/performance/PanamaPerformanceOptimizations.java
public class PanamaPerformanceOptimizations {

    // Method handle caching and optimization
    public static class MethodHandleOptimizer {
        private static final Map<String, MethodHandle> optimizedHandles = new ConcurrentHashMap<>();

        // Create specialized method handles for common patterns
        static {
            try {
                // Pre-create optimized handles for common function signatures
                createOptimizedHandle("func_call_i32", "(JJI)I");
                createOptimizedHandle("func_call_i64", "(JJJ)J");
                createOptimizedHandle("func_call_f32", "(JJF)F");
                createOptimizedHandle("func_call_f64", "(JJD)D");
                createOptimizedHandle("memory_read_bulk", "(JJJI)V");
                createOptimizedHandle("memory_write_bulk", "(JJJI)V");
            } catch (Exception e) {
                throw new RuntimeException("Failed to initialize optimized method handles", e);
            }
        }

        private static void createOptimizedHandle(String name, String signature) throws Exception {
            SymbolLookup lookup = PanamaNativeLibrary.getSymbolLookup();
            MethodHandle handle = lookup.find("wasmtime4j_" + name)
                .map(addr -> Linker.nativeLinker().downcallHandle(addr,
                    FunctionDescriptor.of(parseSignature(signature))))
                .orElseThrow();

            optimizedHandles.put(name, handle);
        }

        public static MethodHandle getOptimizedHandle(String name) {
            return optimizedHandles.get(name);
        }
    }

    // Arena optimization for memory management
    public static class ArenaOptimizer {
        // Pool of reusable confined arenas for temporary operations
        private static final ThreadLocal<Deque<Arena>> arenaPool = ThreadLocal.withInitial(ArrayDeque::new);

        public static Arena borrowConfined() {
            Deque<Arena> pool = arenaPool.get();
            if (!pool.isEmpty()) {
                return pool.pop();
            } else {
                return Arena.ofConfined();
            }
        }

        public static void returnConfined(Arena arena) {
            if (arena.scope().isAlive()) {
                arenaPool.get().push(arena);
            }
        }

        // Optimized string allocation
        public static MemorySegment allocateUtf8String(Arena arena, String str) {
            byte[] bytes = str.getBytes(StandardCharsets.UTF_8);
            MemorySegment segment = arena.allocate(bytes.length + 1);
            segment.copyFrom(MemorySegment.ofArray(bytes));
            segment.set(ValueLayout.JAVA_BYTE, bytes.length, (byte) 0);
            return segment;
        }
    }

    // Memory segment operations optimization
    public static class MemorySegmentOptimizer {
        // Optimized bulk operations
        public static void bulkCopy(MemorySegment dest, long destOffset, byte[] src) {
            MemorySegment srcSegment = MemorySegment.ofArray(src);
            dest.asSlice(destOffset, src.length).copyFrom(srcSegment);
        }

        public static void bulkRead(MemorySegment src, long srcOffset, byte[] dest) {
            MemorySegment destSegment = MemorySegment.ofArray(dest);
            destSegment.copyFrom(src.asSlice(srcOffset, dest.length));
        }

        // Optimized value access
        public static int getInt32(MemorySegment segment, long offset) {
            return segment.get(ValueLayout.JAVA_INT_UNALIGNED, offset);
        }

        public static void setInt32(MemorySegment segment, long offset, int value) {
            segment.set(ValueLayout.JAVA_INT_UNALIGNED, offset, value);
        }
    }
}

// Function call optimization with parameter type specialization
#[no_mangle]
pub extern "C" fn wasmtime4j_func_call_i32(
    func_ptr: *mut c_void,
    store_ptr: *mut c_void,
    param: i32
) -> i32 {
    let func = unsafe { &*(func_ptr as *const wasmtime::Func) };
    let mut store = unsafe { &mut *(store_ptr as *mut wasmtime::Store<()>) };

    let params = [Val::I32(param)];
    let mut results = [Val::I32(0)];

    match func.call(&mut store, &params, &mut results) {
        Ok(_) => {
            if let Val::I32(result) = results[0] {
                result
            } else {
                0
            }
        }
        Err(_) => 0,
    }
}

// Bulk memory operations with optimized paths
#[no_mangle]
pub extern "C" fn wasmtime4j_memory_bulk_read_optimized(
    memory_ptr: *mut c_void,
    store_ptr: *mut c_void,
    offset: u32,
    data_ptr: *mut u8,
    size: u32
) -> bool {
    let memory = unsafe { &*(memory_ptr as *const wasmtime::Memory) };
    let mut store = unsafe { &mut *(store_ptr as *mut wasmtime::Store<()>) };

    if size <= 8192 {
        // Use optimized small read path
        let data_slice = unsafe { std::slice::from_raw_parts_mut(data_ptr, size as usize) };
        match memory.read(&store, offset as usize, data_slice) {
            Ok(_) => true,
            Err(_) => false,
        }
    } else {
        // Use streaming approach for large reads
        memory_bulk_read_streaming(memory, &mut store, offset, data_ptr, size)
    }
}

// Type introspection with caching
static mut TYPE_CACHE: Option<std::collections::HashMap<usize, CachedTypeInfo>> = None;

#[derive(Clone)]
struct CachedTypeInfo {
    imports: Vec<ImportInfo>,
    exports: Vec<ExportInfo>,
    func_types: Vec<FuncTypeInfo>,
}

#[no_mangle]
pub extern "C" fn wasmtime4j_module_get_cached_types(module_ptr: *mut c_void) -> *mut c_void {
    let module = unsafe { &*(module_ptr as *const wasmtime::Module) };
    let module_id = module_ptr as usize;

    unsafe {
        if TYPE_CACHE.is_none() {
            TYPE_CACHE = Some(std::collections::HashMap::new());
        }

        let cache = TYPE_CACHE.as_mut().unwrap();
        let cached_info = cache.entry(module_id).or_insert_with(|| {
            CachedTypeInfo {
                imports: collect_import_info(module),
                exports: collect_export_info(module),
                func_types: collect_func_type_info(module),
            }
        });

        Box::into_raw(Box::new(cached_info.clone())) as *mut c_void
    }
}

// Host function call optimization
#[no_mangle]
pub extern "C" fn wasmtime4j_host_function_call_optimized(
    callback_id: usize,
    params_ptr: *const *const c_void,
    params_count: u32,
    results_ptr: *mut *mut c_void,
    results_capacity: u32
) -> i32 {
    // Optimized host function callback with minimal overhead
    unsafe {
        if let Some(ref registry) = HOST_FUNCTION_REGISTRY {
            if let Some(callback) = registry.functions.get(&callback_id) {
                // Convert parameters with optimized paths
                let mut params = Vec::with_capacity(params_count as usize);
                for i in 0..params_count {
                    let param_ptr = *params_ptr.add(i as usize);
                    params.push(convert_param_optimized(param_ptr));
                }

                // Call function with minimal overhead
                match callback(&params) {
                    Ok(results) => {
                        for (i, result) in results.iter().enumerate() {
                            if i < results_capacity as usize {
                                *results_ptr.add(i) = convert_result_optimized(result);
                            }
                        }
                        results.len() as i32
                    }
                    Err(_) => -1,
                }
            } else {
                -1
            }
        } else {
            -1
        }
    }
}

// src/test/java/ai/tegmentum/wasmtime4j/benchmarks/ComprehensiveBenchmarkSuite.java
@BenchmarkMode(Mode.AverageTime)
@OutputTimeUnit(TimeUnit.NANOSECONDS)
@State(Scope.Benchmark)
@Warmup(iterations = 5, time = 1, timeUnit = TimeUnit.SECONDS)
@Measurement(iterations = 10, time = 1, timeUnit = TimeUnit.SECONDS)
@Fork(1)
public class ComprehensiveBenchmarkSuite {

    @Param({"JNI", "PANAMA"})
    private String runtime;

    @Param({"simple_add", "complex_math", "memory_intensive", "string_processing"})
    private String benchmark;

    private Engine engine;
    private Module module;
    private Store store;
    private Instance instance;
    private WasmFunction function;

    @Setup(Level.Trial)
    public void setup() {
        System.setProperty("wasmtime4j.runtime", runtime.toLowerCase());

        Config config = Config.builder()
            .withOptimizationLevel(OptimizationLevel.SPEED)
            .build();

        engine = Engine.create(config);
        byte[] wasmBytes = BenchmarkResources.getModuleByName(benchmark);
        module = Module.compile(engine, wasmBytes);
        store = Store.create(engine);
        instance = Instance.create(store, module, new ArrayList<>());
        function = instance.getExportedFunction("benchmark_function").orElseThrow();
    }

    @Benchmark
    public Object functionCall() throws Exception {
        return function.call(store, 42, 3.14);
    }

    @Benchmark
    public Object optimizedFunctionCall() throws Exception {
        return PerformanceOptimizer.FunctionCallOptimizer.optimizedCall(function, store, 42, 3.14);
    }

    @Benchmark
    public void memoryAccess() {
        Memory memory = instance.getExportedMemory("memory").orElseThrow();
        byte[] data = new byte[1024];
        memory.write(store, 0, data);
        memory.read(store, 0, data);
    }

    @Benchmark
    public void optimizedMemoryAccess() {
        Memory memory = instance.getExportedMemory("memory").orElseThrow();
        byte[] data = new byte[1024];
        PerformanceOptimizer.MemoryAccessOptimizer.bulkWrite(memory, store, 0, data);
        PerformanceOptimizer.MemoryAccessOptimizer.bulkRead(memory, store, 0, data);
    }

    @Benchmark
    public void typeIntrospection() {
        module.getExports();
        module.getImports();
        module.getFunctionTypes();
    }

    @Benchmark
    public void optimizedTypeIntrospection() {
        PerformanceOptimizer.TypeIntrospectionOptimizer.ModuleTypeCache cache =
            PerformanceOptimizer.TypeIntrospectionOptimizer.getTypeCache(module);
        cache.getExports();
        cache.getImports();
        cache.getFunctionTypes();
    }

    // Performance requirement validation
    @Test
    public void validatePerformanceRequirements() {
        Options opt = new OptionsBuilder()
            .include(this.getClass().getSimpleName())
            .forks(1)
            .build();

        Collection<RunResult> results = new Runner(opt).run();

        // Validate specific performance requirements
        validateHostFunctionOverhead(results);
        validateMemoryOperationPerformance(results);
        validateTypeIntrospectionOverhead(results);
    }

    private void validateHostFunctionOverhead(Collection<RunResult> results) {
        double jniHostFunctionTime = getResult(results, "JNI", "functionCall");
        double panamaHostFunctionTime = getResult(results, "PANAMA", "functionCall");
        double nativeBaselineTime = getNativeBaselineTime();

        double jniOverhead = (jniHostFunctionTime - nativeBaselineTime) / nativeBaselineTime;
        double panamaOverhead = (panamaHostFunctionTime - nativeBaselineTime) / nativeBaselineTime;

        assertTrue(jniOverhead < 0.10, "JNI host function overhead exceeds 10%: " + jniOverhead);
        assertTrue(panamaOverhead < 0.10, "Panama host function overhead exceeds 10%: " + panamaOverhead);
    }
}
