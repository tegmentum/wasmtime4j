## Error Handling Patterns

### Before Integration (Duplicated)
```rust
// In jni_bindings.rs
if engine_ptr.is_null() {
    return Err("Engine pointer is null".into());
}
match wasmtime_error {
    WasmtimeError::Validation(msg) => return Err(format!("Validation: {}", msg)),
    WasmtimeError::Runtime(msg) => return Err(format!("Runtime: {}", msg)), 
    // ... more error handling
}

// In panama_ffi.rs - SIMILAR BUT POTENTIALLY INCONSISTENT
if engine_ptr.is_null() {
    return Err("Null engine pointer".into()); // Different message!
}
match wasmtime_error {
    WasmtimeError::Validation(msg) => return Err(format!("Validation error: {}", msg)), // Different format!
    // ... potentially different error handling
}
```

```rust  
// In both jni_bindings.rs and panama_ffi.rs
let engine = ffi_common::error_handling::validate_pointer(engine_ptr, "engine")?;
let error_info = ffi_common::error_handling::convert_wasmtime_error(wasmtime_error);
return Err(error_info.message);
```
